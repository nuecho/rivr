019/**
020 * Implementation of {@link DialogueChannel} allowing turns to be apparently
021 * exchanged in a regular synchronous fashion while underneath, it is done in
022 * asynchronously.
023 * <p>
024 * This is the most useful implementation of {@link DialogueChannel}. It is used
025 * by the {@link DialogueServlet} to handle different HTTP requests occurring in
026 * separate threads.
027 * <p>
028 * When the controller (such as the {@link DialogueServlet}) uses the
029 * {@link SynchronousDialogueChannel}, it uses the
030 * {@link #start(Dialogue, FirstTurn, Duration, DialogueContext)} method to
031 * start the dialogue and the {@link #doTurn(InputTurn, Duration)} method to
032 * send input turns to the dialogue. In exchange, the
033 * {@link SynchronousDialogueChannel} will return a {@link Step}:
034 * <ul>
035 * <li>{@link OutputTurnStep}: if the dialogue sends an OutputTurn in response</li>
036 * <li>{@link LastTurnStep}: when the dialogue is done</li>
037 * <li>{@link ErrorStep}: if an error occurred following the delivery of the
038 * {@link InputTurn}</li>
039 * </ul>
040 * <p>
041 * <h3>States</h3> The {@link SynchronousDialogueChannel} has states related to
042 * the dialogue:
043 * <ul>
044 * <li>started</li>
045 * <li>done</li>
046 * </ul>
047 * The {@link #isDialogueStarted()} and {@link #isDialogueDone()} methods
048 * correspond to those states. Additionally, the {@link #isDialogueActive()}
049 * method tells if the dialogue is <i>started</i> but not yet <i>done</i>.
050 * <p>
051 * <h3>Time-out values</h3> The {@link SynchronousDialogueChannel} internally
052 * keeps two {@link SynchronousQueue SynchronousQueues}:
053 * <ul>
054 * <li>one for communication of output turn from the {@link Dialogue} to the
055 * controller</li>
056 * <li>one for communication of input turn from the controller to the
057 * {@link Dialogue}</li>
058 * </ul>
059 * The default timeout values used while waiting from controller or dialogue can
060 * be set with {@link #setDefaultReceiveFromControllerTimeout(Duration)} and
061 * {@link #setDefaultReceiveFromDialogueTimeout(Duration)}. These properties are
062 * used if the <code>timeout</code> parameter is <code>null</code> for the
063 * {@link #doTurn(OutputTurn, Duration)},
064 * {@link #start(Dialogue, FirstTurn, Duration, DialogueContext)} and
065 * {@link #doTurn(InputTurn, Duration)} methods.
066 * <p>
067 * If not set, <code>defaultReceiveFromControllerTimeout</code> defaults to <b>5
068 * minutes</b> and <code>defaultReceiveFromDialogueTimeout</code> defaults to
069 * <b>1 minute</b>. This gives 5 minutes for the controller (like the
070 * {@link DialogueServlet}) to provide the {@link InputTurn} and 1 minute for
071 * the controller to provide the next turn or error before a {@link Timeout}
072 * exception is raised.
073 * <p>
074 * <b>Note:</b> There also exists a less important <code>sendTimeout</code>
075 * property which indicates the maximum duration for send operations. A send
076 * operation occurs when the controller sends the input turn to the dialogue and
077 * when the dialogue send the output turn, the last turn or an error to the
078 * controller. Since we expect the dialogue and the controller to already be in
079 * a waiting state when the other thread sends the turn, the send operation
080 * should not block. For this reason, the default value for this property is
081 * <b>5 seconds</b>. This property can be set with the
082 * {@link #setSendTimeout(Duration)} property.
083 * <p>
084 * 
085 * @param <F> type of {@link FirstTurn}
086 * @param <L> type of {@link LastTurn}
087 * @param <O> type of {@link OutputTurn}
088 * @param <I> type of {@link InputTurn}
089 * @param <C> type of {@link DialogueContext}
090 * @see DialogueServlet
091 * @see Step
092 * @see OutputTurnStep
093 * @see LastTurnStep
094 * @see ErrorStep
095 * @author Nu Echo Inc.
096 */
097public final class SynchronousDialogueChannel<I extends InputTurn, O extends OutputTurn, F extends FirstTurn, L extends LastTurn, C extends DialogueContext<I, O>>
098        implements DialogueChannel<I, O> {
099
100    private NamedSynchronousQueue<Step<O, L>> mFromDialogueToController = new NamedSynchronousQueue<Step<O, L>>("dialogue to controller",
101                                                                                                                true);
102    private NamedSynchronousQueue<I> mFromControllerToDialogue = new NamedSynchronousQueue<I>("controller to dialogue",
103                                                                                              true);
104    private Thread mDialogueThread;
105
106    private Duration mSendTimeout = Duration.seconds(5);
107
108    private Duration mDefaultReceiveFromDialogueTimeout = Duration.minutes(1);
109    private Duration mDefaultReceiveFromControllerTimeout = Duration.minutes(5);
110
111    private final List<DialogueChannelListener<I, O>> mListener = new ArrayList<DialogueChannelListener<I, O>>();
112    private Logger mLogger = NOPLogger.NOP_LOGGER;
113
114    private volatile boolean mStopped;
115    private volatile boolean mDialogueStarted;
116    private volatile boolean mDialogueDone;

199    /**
200     * Starts a {@link Dialogue} in a new thread.
201     * 
202     * @param dialogue Dialogue to start. Cannot be <code>null</code>.
203     * @param firstTurn First turn used passed to
204     *            {@link Dialogue#run(FirstTurn, DialogueContext)} method of the
205     *            dialogue. Cannot be <code>null</code>.
206     * @param timeout maximum time allowed to receive the turn from the
207     *            dialogue. If <code>null</code>, uses the
208     *            <code>defaultReceiveFromDialogueTimeout</code> property. A
209     *            value of Duration.ZERO (or equivalent) means to wait forever.
210     * @param context Dialogue context to pass to
211     *            {@link Dialogue#run(FirstTurn, DialogueContext)} method of the
212     *            dialogue. Cannot be <code>null</code>.
213     * @throws Timeout If no result can be obtain from dialogue after delay
214     *             specified by <code>timeout</code> parameter.
215     */
216    public Step<O, L> start(final Dialogue<I, O, F, L, C> dialogue, final F firstTurn, Duration timeout, final C context)
217            throws Timeout, InterruptedException {
218
219        Runnable runnable = new Runnable() {
220            @Override
221            public void run() {
222                mDialogueStarted = true;
223                for (DialogueChannelListener<I, O> listener : mListener) {
224                    listener.onStart(SynchronousDialogueChannel.this);
225                }
226
227                Step<O, L> lastStep;
228                try {
229                    L lastTurn = dialogue.run(firstTurn, context);
230                    lastStep = new LastTurnStep<O, L>(lastTurn);
231                } catch (Throwable throwable) {
232                    mLogger.error("Error in dialogue.", throwable);
233                    lastStep = new ErrorStep<O, L>(throwable);
234                }
235
236                try {
237                    if (!mStopped) {
238                        mLogger.trace("Last step: {}", lastStep);
239                        send(mFromDialogueToController, lastStep, mSendTimeout);
240                    }
241                } catch (Timeout exception) {
242                    throw new RuntimeException("Timeout while sending final result.", exception);
243                } catch (InterruptedException exception) {
244                    mLogger.info("Dialogue interrupted while sending final result.");
245                    Thread.currentThread().interrupt();
246                } finally {
247                    mDialogueDone = true;
248                    mFromDialogueToController = null; // ensure can't receive further output turns from dialogue
249                    mFromControllerToDialogue = null; // ensure can't send further input turns to dialogue
250                    for (DialogueChannelListener<I, O> listener : mListener) {
251                        listener.onStop(SynchronousDialogueChannel.this);
252                    }
253                    mLogger.info("Dialogue ended.");
254                }
255            }
256        };
257
258        mDialogueThread = new Thread(runnable, "Dialogue " + context.getDialogueId());
259        mDialogueThread.start();
260        mLogger.info("Dialogue started.");
261
262        return receive(mFromDialogueToController, timeout);
263    }

324    /**
325     * Performs a turn exchange: the dialogue channel will return the
326     * {@link InputTurn}
327     * 
328     * @param turn The output turn to send. Cannot be <code>null</code>.
329     * @param timeout maximum time allowed to receive the turn from the
330     *            dialogue. If <code>null</code>, uses the
331     *            <code>defaultReceiveFromControllerTimeout</code>. A value of
332     *            Duration.ZERO (or equivalent) means to wait forever.
333     * @return the received {@link InputTurn}. Cannot be <code>null</code>.
334     * @throws Timeout if the dialogue channel has not been able to give the
335     *             InputTurn before <code>timeout</code> parameter.
336     * @throws InterruptedException if the thread has been interrupted while
337     *             waiting for the result.
338     */
339
340    @Override
341    public I doTurn(O turn, Duration timeout) throws Timeout, InterruptedException {
342        verifyState();
343        mLogger.trace("OutputTurn: {}", turn);
344        OutputTurnStep<O, L> turnStep = new OutputTurnStep<O, L>(turn);
345        if (timeout == null) {
346            timeout = mDefaultReceiveFromControllerTimeout;
347        }
348        return exchange(mFromDialogueToController, mFromControllerToDialogue, turnStep, mSendTimeout, timeout);
349    }
350
351    /**
352     * Performs a turn exchange: the dialogue channel will return the Step
353     * 
354     * @param turn the input turn to send to the dialogue
355     * @param timeout maximum time allowed to receive the turn from the
356     *            dialogue. If <code>null</code>, uses the
357     *            <code>defaultReceiveFromDialogueTimeout</code> property. A
358     *            value of Duration.ZERO (or equivalent) means to wait forever.
359     * @throws Timeout If no result can be obtain from dialogue after delay
360     *             specified by <code>timeout</code> parameter.
361     * @return the {@link Step} wrapping the dialogue next step
362     */
363    public Step<O, L> doTurn(I turn, Duration timeout) throws Timeout, InterruptedException {
364        verifyState();
365        mLogger.trace("InputTurn: {}", turn);
366        if (timeout == null) {
367            timeout = mDefaultReceiveFromDialogueTimeout;
368        }
369        return exchange(mFromControllerToDialogue, mFromDialogueToController, turn, mSendTimeout, timeout);
370    }
371
372    private void verifyState() {
373        if (mDialogueThread == null) throw new IllegalStateException("Dialogue is not set");
374        if (!mDialogueThread.isAlive()) throw new IllegalStateException("Dialogue is not started");
375        if (mStopped) throw new IllegalStateException("Dialogue is stopped");
376    }
377
378    private <S, R> R exchange(NamedSynchronousQueue<S> sendQueue,
379                              NamedSynchronousQueue<R> receiveQueue,
380                              S itemToSend,
381                              Duration sendTimeout,
382                              Duration receiveTimeout) throws Timeout, InterruptedException {
383        try {
384            send(sendQueue, itemToSend, sendTimeout);
385            return receive(receiveQueue, receiveTimeout);
386        } catch (InterruptedException interruptedException) {
387            if (mStopped) throw new DialogueChannelStopped();
388            else throw interruptedException;
389        }
390    }
391
392    private static <R> R receive(NamedSynchronousQueue<R> receiveQueue, Duration timeout) throws Timeout,
393            InterruptedException {
394        if (receiveQueue == null) throw new IllegalStateException("Receive queue is closed.");
395        R result = receiveQueue.poll(timeout.getMilliseconds(), TimeUnit.MILLISECONDS);
396        if (result == null)
397            throw new Timeout("Timed-out in receive() after " + timeout + " in [" + receiveQueue.getName() + "]");
398        return result;
399    }
400
401    private static <S> void send(NamedSynchronousQueue<S> sendQueue, S itemToSend, Duration timeout) throws Timeout,
402            InterruptedException {
403        if (sendQueue == null) throw new IllegalStateException("Send queue is closed.");
404        boolean success = sendQueue.offer(itemToSend, timeout.getMilliseconds(), TimeUnit.MILLISECONDS);
405        if (!success) throw new Timeout("Timed-out in send() after " + timeout + " in [" + sendQueue.getName() + "]");
406    }
