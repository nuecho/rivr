005    package com.nuecho.rivr.core.channel.synchronous.step;
006    
007    import com.nuecho.rivr.core.channel.*;
008    import com.nuecho.rivr.core.util.*;
009    
010    /**
011     * @author Nu Echo Inc.
012     */
013    public final class ErrorStep<O extends OutputTurn, L extends LastTurn> implements Step<O, L> {
014    
015        private final Throwable mThrowable;
016    
017        public ErrorStep(Throwable throwable) {
018            Assert.notNull(throwable, "throwable");
019            mThrowable = throwable;
020        }
021    
022        public Throwable getThrowable() {
023            return mThrowable;
024        }
025    
026        @Override
027        public String toString() {
028            return "Error: " + mThrowable.toString();
029        }
030    }
