022/**
023 * Abstract servlet interacting with a web client acting as the controller of a
024 * {@link SynchronousDialogueChannel}.
025 * <p>
026 * This abstract servlet must be extended in order to provide a specific
027 * implementation. For each session,
028 * <ol>
029 * <li>the servlet creates the {@link Session} and place it in the
030 * {@link SessionContainer}</li>
031 * <li>it creates a {@link Dialogue} with a {@link DialogueFactory}</li>
032 * <li>it creates a {@link DialogueContext} with a
033 * {@link DialogueContextFactory}</li>
034 * <li>it creates a {@link SynchronousDialogueChannel} and starts the dialogue
035 * upon initial HTTP request</li>
036 * <li>it renders the various {@link Step steps} from the dialogue channel into
037 * appropriate HTTP responses</li>
038 * <li>it translates HTTP requests into {@link InputTurn InputTurns} using the
039 * {@link InputTurnFactory}
040 * <li>once the dialogue is done, the servlet perform necessary clean-up.</li>
041 * </ol>
042 * <h3>init args</h3> The following servlet initial arguments are supported:
043 * <p/>
044 * <dt>com.nuecho.rivr.core.dialogueTimeout
045 * <dd>Maximum time for dialogue to produce an {@link OutputTurn}. Value
046 * specified must be followed by unit (ms, s, m, h, d, y), e.g. <code>10s</code>
047 * for 10 seconds. Default value: <code>10 s</code></dd></dt>
048 * <p/>
049 * <dt>com.nuecho.rivr.core.sessionTimeout</dt>
050 * <dd>Maximum inactivity time for a session. Value specified must be followed
051 * by unit (ms, s, m, h, d, y), e.g. <code>10s</code> for 10 seconds. Default
052 * value: <code>30 m</code></dd>
053 * <p/>
054 * <dt>com.nuecho.rivr.core.sessionScanPeriod</dt>
055 * <dd>Time between each scan for dead sessions in the session container. Value
056 * specified must be followed by unit (ms, s, m, h, d, y), e.g. <code>10s</code>
057 * for 10 seconds. Default value: <code>2 m</code></dd>
058 * 
059 * @param <F> type of {@link FirstTurn}
060 * @param <L> type of {@link LastTurn}
061 * @param <O> type of {@link OutputTurn}
062 * @param <I> type of {@link InputTurn}
063 * @param <C> type of {@link DialogueContext}
064 * @author Nu Echo Inc.
065 */
066public abstract class DialogueServlet<I extends InputTurn, O extends OutputTurn, F extends FirstTurn, L extends LastTurn, C extends DialogueContext<I, O>>
067        extends HttpServlet {

109    /**
110     * Initializes the servlet. The first thing done in this method is to call
111     * {@link #initDialogueServlet()}. This method is called by the servlet
112     * container.
113     */
114    @Override
115    public final void init() throws ServletException {
116
117        try {
118            initDialogueServlet();
119        } catch (DialogueServletInitializationException exception) {
120            throw new ServletException("Unable to initialize dialogue servlet.s", exception);
121        }
122
123        if (mLoggerFactory == null) {
124            mLoggerFactory = LoggerFactory.getILoggerFactory();
125        }
126
127        ensureFieldIsSet(mInputTurnFactory, "InputTurnFactory");
128        ensureFieldIsSet(mDialogueFactory, "DialogueFactory");
129        ensureFieldIsSet(mDialogueContextFactory, "DialogueContextFactory");
130        ensureFieldIsSet(mErrorHandler, "ErrorHandler");
131
132        Logger logger = mLoggerFactory.getLogger(SESSION_LOGGER_NAME);
133
134        mSessionContainer = new SessionContainer<I, O, F, L, C>(logger,
135                                                                mSessionTimeout,
136                                                                mSessionScanPeriod,
137                                                                SESSION_CONTAINER_NAME);
138
139        Duration sessionScanPeriod = getDuration(INITIAL_ARGUMENT_SESSION_SCAN_PERIOD);
140        if (sessionScanPeriod != null) {
141            setSessionScanPeriod(sessionScanPeriod);
142        }
143
144        Duration sessionTimeout = getDuration(INITIAL_ARGUMENT_SESSION_TIMEOUT);
145        if (sessionTimeout != null) {
146            setSessionTimeout(sessionTimeout);
147        }
148
149        Duration dialogueTimeout = getDuration(INITIAL_ARGUMENT_DIALOGUE_TIMEOUT);
150        if (dialogueTimeout != null) {
151            setDialogueTimeout(dialogueTimeout);
152        }
153
154    }

166    private Duration getDuration(String key) throws ServletException {
167        ServletConfig servletConfig = getServletConfig();
168        String duration = servletConfig.getInitParameter(key);
169        if (duration == null) return null;
170        try {
171            return Duration.parse(duration);
172        } catch (IllegalArgumentException exception) {
173            throw new ServletException("Unable to parse duration for init-arg '" + key + "'", exception);
174        }
175    }

195    protected void renderOutputTurn(O outputTurn,
196                                    HttpServletRequest request,
197                                    final HttpServletResponse response,
198                                    Session<I, O, F, L, C> session) throws IOException, StepRendererException {

278    private void process(HttpServletRequest request, HttpServletResponse response) throws ServletException {
279        Session<I, O, F, L, C> session;
280        try {
281            session = getSession(request);
282            MDC.put(MDC_KEY_DIALOGUE_ID, session.getId());
283        } catch (SessionNotFoundException exception) {
284            throw new ServletException("Cannot find session.", exception);
285        }
286
287        process(request, response, session);
288    }
289
290    private void process(HttpServletRequest request, HttpServletResponse response, Session<I, O, F, L, C> session)
291            throws ServletException {
292        try {
293
294            Step<O, L> step;
295            C dialogueContext = session.getDialogueContext();
296
297            try {
298                if (dialogueContext == null) {
299                    step = startDialogue(request, response, session);
300                } else {
301                    step = continueDialogue(request, response, session);
302                }
303            } catch (Timeout exception) {
304                renderError(exception, request, response, session);
305                return;
306            } catch (InterruptedException exception) {
307                Thread.currentThread().interrupt();
308                renderError(exception, request, response, session);
309                return;
310            }
311
312            if (step instanceof OutputTurnStep) {
313                OutputTurnStep<O, L> outputTurnStep = (OutputTurnStep<O, L>) step;
314                renderOutputTurn(outputTurnStep.getOutputTurn(), request, response, session);
315            } else if (step instanceof LastTurnStep) {
316                LastTurnStep<O, L> lastTurnStep = (LastTurnStep<O, L>) step;
317                renderLastTurn(lastTurnStep.getLastTurn(), request, response, session);
318            } else if (step instanceof ErrorStep) {
319                ErrorStep<O, L> errorStep = (ErrorStep<O, L>) step;
320                Throwable throwable = errorStep.getThrowable();
321                renderError(throwable, request, response, session);
322            }
323        } catch (Exception exception) {
324            throw new ServletException("Error while rendering step.", exception);
325        }
326    }

328    private Step<O, L> continueDialogue(HttpServletRequest request,
329                                        HttpServletResponse response,
330                                        Session<I, O, F, L, C> session) throws ServletException, Timeout,
331            InterruptedException {
332        Assert.notNull(session, "session");
333        I inputTurn = createInputTurn(request, response);
334        SynchronousDialogueChannel<I, O, F, L, C> dialogueChannel = session.getDialogueChannel();
335        Assert.notNull(dialogueChannel, "dialogueChannel");
336        return dialogueChannel.doTurn(inputTurn, mDialogueTimeout);
337    }
338
339    private Step<O, L> startDialogue(HttpServletRequest request,
340                                     HttpServletResponse response,
341                                     Session<I, O, F, L, C> session) throws ServletException, Timeout,
342            InterruptedException {
343        SynchronousDialogueChannel<I, O, F, L, C> dialogueChannel;
344        dialogueChannel = new SynchronousDialogueChannel<I, O, F, L, C>();
345        session.setDialogueChannel(dialogueChannel);
346
347        Logger logger = mLoggerFactory.getLogger(DIALOGUE_LOGGER_NAME);
348        dialogueChannel.setLogger(logger);
349
350        C dialogueContext = createContext(request, session, dialogueChannel, logger);
351
352        DialogueInitializationInfo<I, O, C> initializationInfo;
353        initializationInfo = createInitializationInfo(request, response, dialogueContext);
354        Dialogue<I, O, F, L, C> dialogue;
355        try {
356            dialogue = mDialogueFactory.create(initializationInfo);
357        } catch (DialogueFactoryException exception) {
358            throw new ServletException("Unable to create dialogue.", exception);
359        }
360        F firstTurn = createFirstTurn(request, response);
361        return dialogueChannel.start(dialogue, firstTurn, mDialogueTimeout, dialogueContext);
362    }

364    private C createContext(HttpServletRequest request,
365                            Session<I, O, F, L, C> session,
366                            SynchronousDialogueChannel<I, O, F, L, C> dialogueChannel,
367                            Logger logger) {
368        C dialogueContext = mDialogueContextFactory.createDialogueContext(request,
369                                                                          session.getId(),
370                                                                          dialogueChannel,
371                                                                          logger);
372        session.setDialogueContext(dialogueContext);
373        return dialogueContext;
374    }

382    protected Session<I, O, F, L, C> getSession(HttpServletRequest request) throws SessionNotFoundException {
383        String pathInfo = request.getPathInfo();
384
385        if (pathInfo != null && !pathInfo.equals("/")) {
386            if (pathInfo.startsWith("/")) {
387                pathInfo = pathInfo.substring(1);
388            }
389
390            int firstSlash = pathInfo.indexOf('/');
391            if (firstSlash != -1) {
392                pathInfo = pathInfo.substring(0, firstSlash);
393            }
394
395            return getExistingSession(pathInfo);
396        } else {
397            String sessionId = UUID.randomUUID().toString();
398
399            Session<I, O, F, L, C> session = new Session<I, O, F, L, C>(mSessionContainer, sessionId);
400            mSessionContainer.addSession(session);
401            return session;
402        }
403    }
404
405    protected Session<I, O, F, L, C> getExistingSession(String sessionId) throws SessionNotFoundException {
406        Session<I, O, F, L, C> session = mSessionContainer.getSession(sessionId);
407
408        if (session == null) throw new SessionNotFoundException("Unable to find session [" + sessionId + "]");
409
410        return session;
411    }

429    private static void commitToResponse(final HttpServletResponse response, ServletResponseContent responseContent)
430            throws IOException {
431        ServletOutputStream outputStream = response.getOutputStream();
432        response.setContentType(responseContent.getContentType());
433        responseContent.writeTo(outputStream);
