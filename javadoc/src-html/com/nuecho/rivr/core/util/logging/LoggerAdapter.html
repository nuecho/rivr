011/**
012 * Base {@link Logger} implementation.
013 * 
014 * @author Nu Echo Inc.
015 */
016public abstract class LoggerAdapter implements Logger {
017    /**
018     * To be overridden.
019     * 
020     * @param marker A marker (can be null).
021     * @param logLevel The log level
022     * @param message The message
023     * @param throwable a throwable (can be null)
024     */
025    protected void log(Marker marker, LogLevel logLevel, String message, Throwable throwable) {}
026
027    /**
028     * To be overridden.
029     * 
030     * @param marker A marker (can be null).
031     * @param logLevel The log level
032     * @param format The format string
033     * @param arguments The arguments to format
034     */
035    protected void logWithFormatting(Marker marker, LogLevel logLevel, String format, Object... arguments) {}
036
037    @Override
038    public void trace(String message) {
039        log(null, TRACE, message, null);
040    }

045    }
