014    public abstract class LoggerAdapter implements Logger {
015        /**
016         * To be overridden.
017         * 
018         * @param marker A marker (can be null).
019         * @param logLevel The log level
020         * @param message The message
021         * @param throwable a throwable (can be null)
022         */
023        protected void log(Marker marker, LogLevel logLevel, String message, Throwable throwable) {}
024    
025        /**
026         * To be overridden.
027         * 
028         * @param marker A marker (can be null).
029         * @param logLevel The log level
030         * @param format The format string
031         * @param arguments The arguments to format
032         */
033        protected void logWithFormatting(Marker marker, LogLevel logLevel, String format, Object... arguments) {}
034    
035        @Override

037            log(null, TRACE, message, null);
