005    package com.nuecho.rivr.core.util;
006    
007    import java.util.*;
008    
009    /**
010     * @author Nu Echo Inc.
011     */
012    public final class Assert {
013    
014        private Assert() {
015            //utility class: instantiation forbidden
016        }
017    
018        public static void notNull(Object item, String symbol) {
019            if (item == null) throw new AssertionError(symbol + " should not be null");
020        }
021    
022        public static void notEmpty(String string, String symbol) {
023            notNull(string, symbol);
024            ensure(!string.isEmpty(), symbol);
025        }
026    
027        public static void notEmpty(Collection<?> collection, String symbol) {
028            notNull(collection, symbol);
029            ensure(!collection.isEmpty(), symbol);
030        }
031    
032        public static void between(long min, long value, long max) {
033            if (value < min || value > max)
034                throw new AssertionError("Value should not be in the range [" + min + ", " + max + "]");
035        }
036    
037        public static void between(double min, double value, double max) {
038            if (value < min || value > max)
039                throw new AssertionError("Value should not be in the range [" + min + ", " + max + "]");
040        }
041    
042        public static void notNegative(long value, String symbol) {
043            ensure(value >= 0, symbol + " should not be negative. Value=" + value);
044        }
045    
046        public static void positive(long value, String symbol) {
047            ensure(value > 0, symbol + " should be positive. Value=" + value);
048        }
049    
050        public static void ensure(boolean condition, String message) {
051            if (!condition) throw new AssertionError(message);
052        }
053    
054    }
