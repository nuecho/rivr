004package com.nuecho.rivr.core.test;
005
006import org.slf4j.*;
007
008import com.nuecho.rivr.core.channel.*;
009import com.nuecho.rivr.core.channel.synchronous.*;
010import com.nuecho.rivr.core.channel.synchronous.step.*;
011import com.nuecho.rivr.core.dialogue.*;
012import com.nuecho.rivr.core.servlet.*;
013import com.nuecho.rivr.core.util.*;
014import com.nuecho.rivr.core.util.logging.*;
015
016/**
017 * A dialogue channel implementation for testing. This {@link DialogueChannel}
018 * is intended for unit testing. Test case acts as the controller and interact
019 * with the dialogue channel. Dialogue steps can be obtained and asserted.
020 * 
021 * @author Nu Echo Inc.
022 */
023public abstract class TestDialogueChannel<I extends InputTurn, O extends OutputTurn, F extends FirstTurn, L extends LastTurn, C extends DialogueContext<I, O>>
024        implements DialogueChannel<I, O> {
025
026    private final SynchronousDialogueChannel<I, O, F, L, C> mChannel;
027    private Step<O, L> mLastStep;
028    private final String mName;
029    private Duration mDefaultTimeout;
030
031    private final AccumulatingLog mAccumulatingLog;
032    private Logger mLogger;
033
034    public TestDialogueChannel(String name, Duration defaultTimeout) {
035        mName = name;
036        mChannel = new SynchronousDialogueChannel<I, O, F, L, C>();
037        mAccumulatingLog = new AccumulatingLog(name);
038        setLogger(mAccumulatingLog);
039        mDefaultTimeout = defaultTimeout;
040    }
041
042    @Override
043    public I doTurn(O outputTurn, Duration timeout) throws Timeout, InterruptedException {
044        return mChannel.doTurn(outputTurn, timeout);
045    }
046
047    @Override
048    public void addListener(DialogueChannelListener<I, O> listener) {
049        mChannel.addListener(listener);
050    }
051
052    @Override
053    public void removeListener(DialogueChannelListener<I, O> listener) {
054        mChannel.removeListener(listener);
055    }
056
057    public void getDefaultTimeout(Duration defaultTimeout) {
058        mDefaultTimeout = defaultTimeout;
059    }
060
061    public void setDefaultTimeout(Duration defaultTimeout) {
062        Assert.notNull(defaultTimeout, "defaultTimeout");
063        mDefaultTimeout = defaultTimeout;
064    }
065
066    public void dispose() {
067
068        if (!mChannel.isDialogueStarted()) throw new AssertionError("The dialogue hasn't run.");
069
070        if (mChannel.isDialogueActive()) {
071            stop();
072            mLogger.warn("Test done but dialogue still active.");
073        }
074
075        mAccumulatingLog.clear();
076    }
077
078    public final void stop() {
079        if (!mChannel.isDialogueActive()) throw new IllegalStateException("Dialogue not active.");
080        try {
081            mChannel.stop(Duration.ZERO);
082        } catch (InterruptedException exception) {
083            mLogger.warn("Interrupted.");
084            Thread.currentThread().interrupt();
085        }
086    }
087
088    public final Step<O, L> startDialogue(Dialogue<I, O, F, L, C> dialogue, F firstTurn, C dialogueContext) {
089        return startDialogue(dialogue, firstTurn, mDefaultTimeout, dialogueContext);
090    }
091
092    public final Step<O, L> startDialogue(Dialogue<I, O, F, L, C> dialogue,
093                                          F firstTurn,
094                                          Duration timeout,
095                                          C dialogueContext) {
096        if (mChannel.isDialogueActive()) throw new IllegalStateException("Dialogue already started.");
097
098        try {
099            mLastStep = mChannel.start(dialogue, firstTurn, timeout, dialogueContext);
100            return mLastStep;
101        } catch (Exception exception) {
102            AssertionError assertionError = new AssertionError("Error during start.");
103            assertionError.initCause(exception);
104            throw assertionError;
105        }
106    }
107
108    public final Step<O, L> processInputTurn(I inputTurn) {
109        return processInputTurn(inputTurn, mDefaultTimeout);
110    }
111
112    public final Step<O, L> processInputTurn(I inputTurn, Duration timeout) {
113
114        if (!mChannel.isDialogueActive()) throw new IllegalStateException("Dialogue not started.");
115
116        try {
117            mLastStep = mChannel.doTurn(inputTurn, timeout);
118            return mLastStep;
119        } catch (Exception exception) {
120            AssertionError assertionError = new AssertionError("Error during start.");
121            assertionError.initCause(exception);
122            throw assertionError;
123        }
124    }
125
126    public Logger getLogger() {
127        return mLogger;
128    }
129
130    public Step<O, L> getLastStep() {
131        return mLastStep;
132    }
133
134    public final O getLastStepAsOutputTurn() {
135        if (mLastStep instanceof OutputTurnStep) {
136            OutputTurnStep<O, L> turnStep = (OutputTurnStep<O, L>) mLastStep;
137            return turnStep.getOutputTurn();
138        }
139
140        throw new AssertionError("Last output turn was not an output turn step. Last step=" + mLastStep);
141    }
142
143    public final L getLastStepAsLastTurn() {
144        if (mLastStep instanceof LastTurnStep) {
145            LastTurnStep<O, L> turnStep = (LastTurnStep<O, L>) mLastStep;
146            return turnStep.getLastTurn();
147        }
148
149        throw new AssertionError("Last output turn was not a last turn step. Last step=" + mLastStep);
150    }
151
152    public final Throwable getLastStepAsError() {
153        if (mLastStep instanceof ErrorStep) {
154            ErrorStep<O, L> errorStep = (ErrorStep<O, L>) mLastStep;
155            return errorStep.getThrowable();
156        }
157
158        throw new AssertionError("Last output turn was not an error step. Last step=" + mLastStep);
159    }
160
161    public final SynchronousDialogueChannel<I, O, F, L, C> getChannel() {
162        return mChannel;
163    }
164
165    public final void dumpLogs() {
166        privateDumpLog(prefixLog(new StandardOutputLogger(LogLevel.ALL)));
167    }
168
169    public final void dumpLogs(LogLevel logLevel) {
170        privateDumpLog(prefixLog(new StandardOutputLogger(logLevel)));
171    }
172
173    private void privateDumpLog(Logger consoleLog) {
174        mAccumulatingLog.flushTo(consoleLog);
175        StackTraceElement callerStackTraceElement = getStackTraceElement(3);
176        StackTraceElement currentStackTraceElement = getStackTraceElement(2);
177        consoleLog.info(currentStackTraceElement.getMethodName()
178                        + "() called at "
179                        + callerStackTraceElement.getClassName()
180                        + "."
181                        + callerStackTraceElement.getMethodName()
182                        + "("
183                        + callerStackTraceElement.getFileName()
184                        + ":"
185                        + callerStackTraceElement.getLineNumber()
186                        + ")");
187
188        setLogger(consoleLog);
189    }
190
191    private void setLogger(Logger log) {
192        mLogger = log;
193        mChannel.setLogger(mLogger);
194    }
195
196    private static StackTraceElement getStackTraceElement(int index) {
197        return new Exception().getStackTrace()[index];
198    }
199
200    private PrefixedLogger prefixLog(Logger log) {
201        return new PrefixedLogger("[" + mName + "] ", log);
202    }
203
204}
