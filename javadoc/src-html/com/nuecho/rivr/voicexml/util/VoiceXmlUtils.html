084        private static final Set<String> RESERVED = new HashSet<String>(Arrays.asList(WORDS));
085    
086        /**
087         * Checks if a name is a valid VoiceXML identifier as per VoiceXML 2.0
088         * specification (section 5.1): <blockquote>The variable naming convention
089         * is as in ECMAScript, but names beginning with the underscore character
090         * ("_") and names ending with a dollar sign ("$") are reserved for internal
091         * use. VoiceXML variables, including form item variables, must not contain
092         * ECMAScript reserved words. They must also follow ECMAScript rules for
093         * referential correctness. For example, variable names must be unique and
094         * their declaration must not include a dot - "var x.y" is an illegal
095         * declaration in ECMAScript. Variable names which violate naming
096         * conventions or ECMAScript rules cause an 'error.semantic' event to be
097         * thrown. </blockquote>
098         * 
099         * @param name the name to be tested
100         * @return <code>true</code> if and only if the name is a valid VoiceXML
101         *         identifier.
102         */
103    
104        public static boolean isValidIdentifierName(String name) {
105            Assert.notNull(name, "name");
106            return ALLOWED.matcher(name).matches()
107                   && !RESERVED.contains(name)
108                   && !name.startsWith("_")
109                   && !name.startsWith("$");
110        }
