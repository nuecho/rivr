213    public static Document createDocument(VoiceXmlDialogueContext voiceXmlDialogueContext) {
214        Element vxmlElement = createVoiceXmlDocumentRoot(voiceXmlDialogueContext);
215
216        String rootDocumentPath = voiceXmlDialogueContext.getContextPath()
217                                  + voiceXmlDialogueContext.getServletPath()
218                                  + VoiceXmlDialogueServlet.ROOT_PATH
219                                  + voiceXmlDialogueContext.getDialogueId();
220
221        vxmlElement.setAttribute(APPLICATION_ATTRIBUTE, rootDocumentPath);
222
223        StringBuilder script = new StringBuilder();
224        script.append(RIVR_SCOPE_OBJECT)
225              .append(".")
226              .append(LOCAL_ERROR_HANDLING_PROPERTY)
227              .append(" = ")
228              .append(FALSE)
229              .append("; ");
230
231        script.append(RIVR_INPUT_TURN_SCOPE_OBJECT).append(" = {};");
232        createScript(vxmlElement, script.toString());
233
234        processFetchRendering(voiceXmlDialogueContext, vxmlElement);
235
236        processProperties(voiceXmlDialogueContext, vxmlElement);
237
238        return vxmlElement.getOwnerDocument();
239    }

250    private static void processFetchRendering(VoiceXmlDialogueContext voiceXmlDialogueContext, Element vxmlElement) {
251        DefaultFetchConfiguration defaultFetchConfiguration = voiceXmlDialogueContext.getFetchConfiguration();
252        if (defaultFetchConfiguration == null) return;
253
254        String fetchAudioLocation = defaultFetchConfiguration.getDefaultFetchAudio();
255        if (fetchAudioLocation != null) {
256            addFetchAudioProperty(vxmlElement, fetchAudioLocation);
257        }
258
259        addDurationProperty(vxmlElement, FETCH_TIMEOUT_PROPERTY, defaultFetchConfiguration.getDefaultFetchTimeout());
260
261        applyResourceDefaultFetchConfiguration(vxmlElement,
262                                               defaultFetchConfiguration.getDefaultAudioFetchConfiguration());
263        applyResourceDefaultFetchConfiguration(vxmlElement,
264                                               defaultFetchConfiguration.getDefaultGrammarFetchConfiguration());
265        applyResourceDefaultFetchConfiguration(vxmlElement,
266                                               defaultFetchConfiguration.getDefaultObjectFetchConfiguration());
267        applyResourceDefaultFetchConfiguration(vxmlElement,
268                                               defaultFetchConfiguration.getDefaultScriptFetchConfiguration());
269    }

300    public static void createPrompt(String language,
301                                    Element parent,
302                                    VoiceXmlDialogueContext voiceXmlDialogueContext,
303                                    Boolean bargeIn,
304                                    List<? extends AudioItem> audioItems) throws VoiceXmlDocumentRenderingException {
305
306        Element promptElement = DomUtils.appendNewElement(parent, PROMPT_ATTRIBUTE);
307        setBooleanAttribute(promptElement, BARGE_IN_PROPERTY, bargeIn);
308
309        String documentLanguage = voiceXmlDialogueContext.getLanguage();
310        if (language != null && !language.equals(documentLanguage)) {
311            promptElement.setAttribute(XML_LANGUAGE_ATTRIBUTE, language);
312        }
313
314        processAudioItems(audioItems, promptElement);
315    }

317    public static void processAudioItems(List<? extends AudioItem> audioItems, Element promptElement)
318            throws VoiceXmlDocumentRenderingException {
319
320        boolean lastItemWasText = false;
321        for (AudioItem audioItem : audioItems) {
322            if (audioItem instanceof AudioFile) {
323                AudioFile audioFile = (AudioFile) audioItem;
324                Element audioElement = DomUtils.appendNewElement(promptElement, AUDIO_ELEMENT);
325
326                //src and expr are mutually exclusive, this is enforced in AudioFile.
327                setAttribute(audioElement, SRC_ATTRIBUTE, audioFile.getLocation());
328                setAttribute(audioElement, EXPR_ATTRIBUTE, audioFile.getExpression());
329
330                SpeechSynthesis alternate = audioFile.getAlternate();
331
332                if (alternate != null) {
333                    if (alternate.isSsml()) {
334                        DocumentFragment ssmlNodes = alternate.getDocumentFragment();
335                        audioElement.appendChild(audioElement.getOwnerDocument().importNode(ssmlNodes, true));
336                        lastItemWasText = false;
337                    } else {
338                        DomUtils.appendNewText(audioElement, alternate.getText());
339                        lastItemWasText = true;
340                    }
341                }
342
343                applyRessourceFetchConfiguration(audioElement, audioFile.getFetchConfiguration());
344            } else if (audioItem instanceof SpeechSynthesis) {
345                SpeechSynthesis speechSynthesis = (SpeechSynthesis) audioItem;
346
347                if (speechSynthesis.isSsml()) {
348                    NodeList childNodes = speechSynthesis.getDocumentFragment().getChildNodes();
349                    Document document = promptElement.getOwnerDocument();
350
351                    for (int childNodeIndex = 0; childNodeIndex < childNodes.getLength(); childNodeIndex++) {
352                        Node node = document.importNode(childNodes.item(childNodeIndex).cloneNode(true), true);
353                        promptElement.appendChild(node);
354                    }
355                    lastItemWasText = false;
356                } else {
357                    String spaceIfRequired = lastItemWasText ? " " : "";
358                    DomUtils.appendNewText(promptElement, spaceIfRequired + speechSynthesis.getText());
359                    lastItemWasText = true;
360                }
361            } else if (audioItem instanceof Pause) {
362                Pause pause = (Pause) audioItem;
363                Element breakElement = DomUtils.appendNewElement(promptElement, BREAK_ELEMENT);
364                setDurationAttribute(breakElement, TIME_ATTRIBUTE, pause.getDuration());
365                lastItemWasText = false;
366            } else if (audioItem instanceof Mark) {
367                Mark mark = (Mark) audioItem;
368                Element markElement = DomUtils.appendNewElement(promptElement, MARK_ELEMENT);
369                markElement.setAttribute(NAME_ATTRIBUTE, mark.getName());
370                lastItemWasText = false;
371            } else throw new VoiceXmlDocumentRenderingException("Cannot handle prompt rendering element of type '"
372                                                                + audioItem.getClass()
373                                                                + "'");
374        }
375    }

451    public static String createEcmaScriptStringLiteral(String content) {
452        if (content == null) return "undefined";
453
454        //believe it: this creates \\ for each \
455        content = ECMASCRIPT_BACKSLASH_PATTERN.matcher(content).replaceAll("\\\\\\\\");
456
457        content = ECMASCRIPT_QUOTE_PATTERN.matcher(content).replaceAll("\\\\'");
458        content = ECMASCRIPT_LINEFEED_PATTERN.matcher(content).replaceAll("\\u000a");
459        content = ECMASCRIPT_CARRIAGE_RETURN_PATTERN.matcher(content).replaceAll("\\u000d");
460        content = ECMASCRIPT_LINE_SEPARATOR_PATTERN.matcher(content).replaceAll("\\u2028");
461        content = ECMASCRIPT_PARAGRAPH_SEPARATOR_PATTERN.matcher(content).replaceAll("\\u2029");
462
463        StringBuilder out = new StringBuilder();
464        out.append("'").append(content).append("'");
465        return out.toString();
466    }

603
604        for (String propertyName : dtmfRecognition.getPropertyNames()) {
605            addProperty(formItemElement, propertyName, dtmfRecognition.getProperty(propertyName));
606        }

638    public static void renderGrammars(List<? extends GrammarItem> grammarItems, Element parent, String mode)
639            throws VoiceXmlDocumentRenderingException {
640        for (GrammarItem grammarItem : grammarItems) {
641            Element grammarElement;
642
643            if (grammarItem instanceof GrammarReference) {
644                grammarElement = DomUtils.appendNewElement(parent, GRAMMAR_ELEMENT);
645                GrammarReference grammarReference = (GrammarReference) grammarItem;
646
647                VoiceXmlDomUtil.renderGrammarReference(grammarReference, grammarElement);
648            } else if (grammarItem instanceof InlineStringGrammar) {
649                grammarElement = DomUtils.appendNewElement(parent, GRAMMAR_ELEMENT);
650                InlineStringGrammar inlineStringGrammar = (InlineStringGrammar) grammarItem;
651                VoiceXmlDomUtil.renderInlineStringGrammar(grammarElement, inlineStringGrammar);
652            } else if (grammarItem instanceof InlineXmlGrammar) {
653                InlineXmlGrammar inlineXmlGrammar = (InlineXmlGrammar) grammarItem;
654                Document document = inlineXmlGrammar.getDocument();
655                Element documentRootElement = document.getDocumentElement();
656                if (!documentRootElement.getTagName().equals(GRAMMAR_ELEMENT))
657                    throw new VoiceXmlDocumentRenderingException("XML grammar root element must be 'gramar'");
658
659                grammarElement = (Element) parent.getOwnerDocument().importNode(documentRootElement, true);
660                parent.appendChild(grammarElement);
661            } else throw new AssertionError("Unsupported grammar item: " + grammarItem);
662
663            Double weight = grammarItem.getWeight();
664            if (weight != null) {
665                DecimalFormat weigthDecimalFormat = new DecimalFormat("#.##########");
666                double weightDoubleValue = weight.doubleValue();
667                grammarElement.setAttribute(WEIGHT_ATTRIBUTE, weigthDecimalFormat.format(weightDoubleValue));
668            }
669
670            setAttribute(grammarElement, TYPE_ATTRIBUTE, grammarItem.getMediaType());
671            setAttribute(grammarElement, MODE_ATTRIBUTE, mode);
672        }
673    }

706    public static void addNamelist(Element parentElement, Element namelistHolderElement, VariableList variables) {
707        if (variables == null) return;
708        if (variables.isEmpty()) return;
709
710        List<String> variableNames = new ArrayList<String>();
711        for (Entry<String, String> entry : variables) {
712            variableNames.add(entry.getKey());
713            createVarElement(parentElement, entry.getKey(), entry.getValue());
714        }
715
716        namelistHolderElement.setAttribute(NAME_LIST_ATTRIBUTE, StringUtils.join(variableNames, " "));
717    }
