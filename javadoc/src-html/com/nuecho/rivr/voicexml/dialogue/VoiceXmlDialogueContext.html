005package com.nuecho.rivr.voicexml.dialogue;
006
007import java.util.*;
008
009import org.slf4j.*;
010
011import com.nuecho.rivr.core.channel.*;
012import com.nuecho.rivr.core.dialogue.*;
013import com.nuecho.rivr.core.util.*;
014import com.nuecho.rivr.voicexml.rendering.voicexml.errorhandling.*;
015import com.nuecho.rivr.voicexml.turn.input.*;
016import com.nuecho.rivr.voicexml.turn.output.*;
017import com.nuecho.rivr.voicexml.turn.output.fetch.*;
018
019/**
020 * VoiceXML specialization of {@link DialogueContext}. In addition to properties
021 * provided by a {@link DialogueContext} (like <code>dialogueChannel</code>,
022 * <code>dialogueId</code>, and <code>logger</code>), the
023 * {@link VoiceXmlDialogueContext} exposes:
024 * <ul>
025 * <li>servlet information used to build URI of some resources.:</li>
026 * <ul>
027 * <li><code>servletPath</code></li>
028 * <li><code>contextPath</code></li>
029 * </ul>
030 * <li>VoiceXML properties (for each generated document)</li> <li>VoiceXML
031 * default language (for each generated document)</li> <li>
032 * <code>turnIndex</code></li> </ul>
033 * 
034 * @author Nu Echo Inc.
035 */
036public final class VoiceXmlDialogueContext implements DialogueContext<VoiceXmlInputTurn, VoiceXmlOutputTurn> {
037
038    private String mLanguage;
039
040    private DefaultFetchConfiguration mDefaultFetchConfiguration = new DefaultFetchConfiguration();
041    private final Map<String, String> mProperties = new HashMap<String, String>();
042
043    private String mDialogueId;
044
045    private final String mContextPath;
046    private final String mServletPath;
047    private int mTurnIndex;
048
049    private final DialogueChannel<VoiceXmlInputTurn, VoiceXmlOutputTurn> mDialogueChannel;
050
051    private final Logger mLogger;
052
053    private FatalErrorFormFactory mFatalErrorFormFactory = new ExitFatalErrorFormFactory();
054
055    public VoiceXmlDialogueContext(DialogueChannel<VoiceXmlInputTurn, VoiceXmlOutputTurn> dialogueChannel,
056                                   Logger logger,
057                                   String dialogueId,
058                                   String contextPath,
059                                   String servletPath) {
060        mDialogueChannel = dialogueChannel;
061        mLogger = logger;
062        mDialogueId = dialogueId;
063        mContextPath = contextPath;
064        mServletPath = servletPath;
065    }
066
067    public void setDialogueId(String dialogueId) {
068        mDialogueId = dialogueId;
069    }
070
071    @Override
072    public String getDialogueId() {
073        return mDialogueId;
074    }
075
076    public String getContextPath() {
077        return mContextPath;
078    }
079
080    public String getServletPath() {
081        return mServletPath;
082    }
083
084    public String getLanguage() {
085        return mLanguage;
086    }
087
088    public void setLanguage(String language) {
089        mLanguage = language;
090    }
091
092    public DefaultFetchConfiguration getFetchConfiguration() {
093        return mDefaultFetchConfiguration;
094    }
095
096    public void setFetchConfiguration(DefaultFetchConfiguration defaultFetchConfiguration) {
097        mDefaultFetchConfiguration = defaultFetchConfiguration;
098    }
099
100    public Map<String, String> getProperties() {
101        return mProperties;
102    }
103
104    public void incrementTurnIndex() {
105        mTurnIndex++;
106    }
107
108    public int getTurnIndex() {
109        return mTurnIndex;
110    }
111
112    @Override
113    public DialogueChannel<VoiceXmlInputTurn, VoiceXmlOutputTurn> getDialogueChannel() {
114        return mDialogueChannel;
115    }
116
117    @Override
118    public Logger getLogger() {
119        return mLogger;
120    }
121
122    public void setFatalErrorFormFactory(FatalErrorFormFactory fatalErrorFormFactory) {
123        mFatalErrorFormFactory = fatalErrorFormFactory;
124    }
125
126    public FatalErrorFormFactory getFatalErrorFormFactory() {
127        return mFatalErrorFormFactory;
128    }
129
130    @Override
131    public String toString() {
132        ToStringBuilder builder = new ToStringBuilder(this);
133        builder.appendItem("mLanguage", mLanguage);
134        builder.appendItem("mDefaultFetchConfiguration", mDefaultFetchConfiguration);
135        builder.appendItem("mProperties", mProperties);
136        builder.appendItem("mDialogueId", mDialogueId);
137        builder.appendItem("mContextPath", mContextPath);
138        builder.appendItem("mServletPath", mServletPath);
139        builder.appendItem("mTurnIndex", mTurnIndex);
140        builder.appendItem("mFatalErrorFormFactory", mFatalErrorFormFactory);
141        return builder.getString();
142    }
143
144}
