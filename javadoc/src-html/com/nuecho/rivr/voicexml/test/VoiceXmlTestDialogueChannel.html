005package com.nuecho.rivr.voicexml.test;
006
007import java.util.*;
008
009import javax.json.*;
010
011import com.nuecho.rivr.core.channel.*;
012import com.nuecho.rivr.core.channel.synchronous.step.*;
013import com.nuecho.rivr.core.test.*;
014import com.nuecho.rivr.core.util.*;
015import com.nuecho.rivr.voicexml.dialogue.*;
016import com.nuecho.rivr.voicexml.turn.first.*;
017import com.nuecho.rivr.voicexml.turn.input.*;
018import com.nuecho.rivr.voicexml.turn.last.*;
019import com.nuecho.rivr.voicexml.turn.output.*;
020import com.nuecho.rivr.voicexml.util.json.*;
021
022/**
023 * Implementation of {@link VoiceXmlDialogueChannel} intended for unit testing.
024 * This class allows a unit test method to be the controller of the dialogue
025 * channel by simulating results from the VoiceXML platform (i.e. DTMF
026 * recognition, events, recording, etc).
027 * <p>
028 * In addition to all methods of {@link TestDialogueChannel}, this class adds
029 * VoiceXML-related functions.
030 * 
031 * @author Nu Echo Inc.
032 */
033public class VoiceXmlTestDialogueChannel
034        extends
035        TestDialogueChannel<VoiceXmlInputTurn, VoiceXmlOutputTurn, VoiceXmlFirstTurn, VoiceXmlLastTurn, VoiceXmlDialogueContext> {
036
037    private static final String CONFIDENCE_PROPERTY = "confidence";
038    private static final String INTERPRETATION_PROPERTY = "interpretation";
039    private static final String INPUTMODE_PROPERTY = "inputmode";
040    private static final String UTTERANCE_PROPERTY = "utterance";
041
042    public VoiceXmlTestDialogueChannel(String name, Duration defaultTimeout) {
043        super(name, defaultTimeout);
044    }
045
046    public final Interaction getLastInteraction() {
047        OutputTurn lastOutputTurn = getLastStepAsOutputTurn();
048        if (lastOutputTurn instanceof Interaction) return (Interaction) getLastStepAsOutputTurn();
049        throw new AssertionError("Last output turn was not an interaction. Last step=" + lastOutputTurn);
050    }
051
052    public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processDtmfRecognition(String dtmfString) {
053        return processDtmfRecognition(dtmfString, JsonValue.NULL, null);
054    }
055
056    public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processDtmfRecognition(String dtmfString,
057                                                                                   JsonValue interpretation,
058                                                                                   MarkInfo markInfo) {
059        JsonArrayBuilder builder = JsonUtils.createArrayBuilder();
060        JsonUtils.add(builder, createDtmfHypothesis(dtmfString, interpretation));
061        return processRecognition(new RecognitionInfo(builder.build(), markInfo));
062    }
063
064    public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processRecognition(JsonArray recognitionResult) {
065        return processRecognition(new RecognitionInfo(recognitionResult, null));
066    }
067
068    public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processRecognition(RecognitionInfo recognitionInfo) {
069        VoiceXmlInputTurn inputTurn = new VoiceXmlInputTurn();
070        inputTurn.setRecognitionInfo(recognitionInfo);
071        return processInputTurn(inputTurn);
072    }
073
074    public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processScript(JsonValue value) {
075        return processValue(value);
076    }
077
078    public Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processValue(JsonValue value) {
079        VoiceXmlInputTurn inputTurn = new VoiceXmlInputTurn();
080        inputTurn.setJsonValue(value);
081        return processInputTurn(inputTurn);
082    }
083
084    public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processRecording(RecordingInfo recordingInfo) {
085        VoiceXmlInputTurn inputTurn = new VoiceXmlInputTurn();
086        inputTurn.setRecordingInfo(recordingInfo);
087        return processInputTurn(inputTurn);
088    }
089
090    public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processNoAction() {
091        return processInputTurn(new VoiceXmlInputTurn());
092    }
093
094    public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processNoMatch() {
095        return processEvent(VoiceXmlEvent.NO_MATCH);
096    }
097
098    public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processNoInput() {
099        return processEvent(VoiceXmlEvent.NO_INPUT);
100    }
101
102    public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processMaxSpeechTimeout() {
103        return processEvent(VoiceXmlEvent.MAX_SPEECH_TIMEOUT);
104    }
105
106    public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processHangup() {
107        return processEvent(VoiceXmlEvent.CONNECTION_DISCONNECT_HANGUP);
108    }
109
110    public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processPlatformError() {
111        return processEvent(VoiceXmlEvent.ERROR);
112    }
113
114    public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processTransferResult(TransferStatusInfo transferStatusInfo) {
115        VoiceXmlInputTurn inputTurn = new VoiceXmlInputTurn();
116        inputTurn.setTransferResult(transferStatusInfo);
117        return processInputTurn(inputTurn);
118    }
119
120    public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processTransferInvalidDestinationResult() {
121        return processEvent(VoiceXmlEvent.ERROR_CONNECTION_BAD_DESTINATION);
122    }
123
124    public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processTransferDisconnect() {
125        return processEvent(VoiceXmlEvent.CONNECTION_DISCONNECT_TRANSFER);
126    }
127
128    public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processEvent(String event) {
129        VoiceXmlInputTurn inputTurn = new VoiceXmlInputTurn();
130        inputTurn.setEvents(createSingleEventList(event));
131        return processInputTurn(inputTurn);
132    }
133
134    private ArrayList<VoiceXmlEvent> createSingleEventList(String eventName) {
135        VoiceXmlEvent event = new VoiceXmlEvent(eventName, null);
136        return new ArrayList<VoiceXmlEvent>(Collections.singleton(event));
137    }
138
139    private JsonObject createDtmfHypothesis(String dtmfString, JsonValue interpretation) {
140        JsonObjectBuilder builder = JsonUtils.createObjectBuilder();
141        JsonUtils.add(builder, UTTERANCE_PROPERTY, dtmfString);
142        JsonUtils.add(builder, INPUTMODE_PROPERTY, "dtmf");
143        JsonUtils.add(builder, INTERPRETATION_PROPERTY, interpretation);
144        builder.add(CONFIDENCE_PROPERTY, 1.0);
145        return builder.build();
146    }
147}
