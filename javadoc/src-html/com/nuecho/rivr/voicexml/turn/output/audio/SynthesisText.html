004    package com.nuecho.rivr.voicexml.turn.output.audio;
005    
006    import javax.json.*;
007    
008    import org.w3c.dom.*;
009    
010    import com.nuecho.rivr.core.util.*;
011    import com.nuecho.rivr.voicexml.util.json.*;
012    
013    /**
014     * A <code>SynthesisText</code> represents a text that will be spoken using a
015     * synthesis voice. It can be either a string of text or a SSML document.
016     * 
017     * @author Nu Echo Inc.
018     * @see <a href="http://www.w3.org/TR/speech-synthesis/">http://www.w3.org/TR/speech-synthesis/</a>
019     */
020    public final class SynthesisText extends AudioItem {
021        public static final String SYNTHESIS_TEXT_ELEMENT_TYPE = "synthesisText";
022        private static final String TEXT_PROPERTY = "text";
023        private static final String SSML_PROPERTY = "ssml";
024    
025        private final String mText;
026        private final DocumentFragment mDocumentFragment;
027    
028        /**
029         * @param text The synthesis text. Not empty.
030         */
031        public SynthesisText(String text) {
032            Assert.notEmpty(text, "text");
033            mText = text;
034            mDocumentFragment = null;
035        }
036    
037        /**
038         * @param documentFragment The SSML document. Not null.
039         */
040        public SynthesisText(DocumentFragment documentFragment) {
041            Assert.notNull(documentFragment, "documentFragment");
042            mDocumentFragment = documentFragment;
043            mText = null;
044        }
045    
046        @Override
047        public String getElementType() {
048            return SYNTHESIS_TEXT_ELEMENT_TYPE;
049        }
050    
051        public String getText() {
052            return mText;
053        }
054    
055        public DocumentFragment getDocumentFragment() {
056            return mDocumentFragment;
057        }
058    
059        public boolean isSsml() {
060            return mDocumentFragment != null;
061        }
062    
063        @Override
064        protected void addJsonProperties(JsonObjectBuilder builder) {
065            if (isSsml()) {
066                JsonUtils.addXmlNodeProperty(builder, SSML_PROPERTY, "SSML fragment", mDocumentFragment);
067            } else {
068                builder.addNull(SSML_PROPERTY);
069            }
070    
071            JsonUtils.add(builder, TEXT_PROPERTY, mText);
072        }
073    
074        @Override
075        public int hashCode() {
076            final int prime = 31;
077            int result = 1;
078            result = prime * result + ((mDocumentFragment == null) ? 0 : mDocumentFragment.hashCode());
079            result = prime * result + ((mText == null) ? 0 : mText.hashCode());
080            return result;
081        }
082    
083        @Override
084        public boolean equals(Object obj) {
085            if (this == obj) return true;
086            if (obj == null) return false;
087            if (getClass() != obj.getClass()) return false;
088            SynthesisText other = (SynthesisText) obj;
089            if (mDocumentFragment == null) {
090                if (other.mDocumentFragment != null) return false;
091            } else if (!mDocumentFragment.equals(other.mDocumentFragment)) return false;
092            if (mText == null) {
093                if (other.mText != null) return false;
094            } else if (!mText.equals(other.mText)) return false;
095            return true;
096        }
097    }
