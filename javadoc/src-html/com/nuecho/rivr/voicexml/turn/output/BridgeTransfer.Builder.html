008import javax.json.*;
009

012import com.nuecho.rivr.core.util.*;

016/**
017 * A {@link BridgeTransfer} is a {@link SupervisedTransfer} that connects the
018 * caller to the callee in a full duplex conversation.
019 * 
020 * @author Nu Echo Inc.
021 * @see <a
022 *      href="http://www.w3.org/TR/voicexml20/#dml2.3.7.2">http://www.w3.org/TR/voicexml20/#dml2.3.7.2</a>
023 */
024public class BridgeTransfer extends SupervisedTransfer {
025    private static final String BRIDGE_TRANSFER_TYPE = "bridge";
026
027    private static final String MAXIMUM_TIME_PROPERTY_NAME = "maximumTime";
028
029    private Duration mMaximumTime;
030
031    /**
032     * @param name The name of this turn. Not empty.
033     * @param destination The URI of the destination (telephone, IP telephony
034     *            address). Not empty.
035     */
036    public BridgeTransfer(String name, String destination) {
037        super(name, destination);
038    }
039
040    /**
041     * @param maximumTime The time that the call is allowed to last.
042     *            <code>null</code> to use the VoiceXML platform default.
043     */
044    public final void setMaximumTime(Duration maximumTime) {
045        mMaximumTime = maximumTime;
046    }
047
048    public final Duration getMaximumTime() {
049        return mMaximumTime;
050    }
051
052    @Override
053    protected final String getTransferType() {
054        return BRIDGE_TRANSFER_TYPE;
055    }
056
057    @Override
058    protected void customizeTransferElement(Element transferElement) throws VoiceXmlDocumentRenderingException {
059        setDurationAttribute(transferElement, MAXTIME_ATTRIBUTE, mMaximumTime);
060    }
061
062    @Override
063    protected void addTurnProperties(JsonObjectBuilder builder) {
064        super.addTurnProperties(builder);
065        JsonUtils.addDurationProperty(builder, MAXIMUM_TIME_PROPERTY_NAME, mMaximumTime);
066    }
067
068    /**
069     * Builder used to ease the creation of instances of {@link BridgeTransfer}.
070     */
071    public static class Builder extends SupervisedTransfer.Builder {
072
073        private Duration mMaximumTime;
074
075        public Builder(String name) {
076            super(name);
077        }
078
079        public Builder setMaximumDuration(Duration maximumTime) {
080            mMaximumTime = maximumTime;
081            return this;
082        }
083
084        public BridgeTransfer build() {
085            BridgeTransfer bridgeTransfer = new BridgeTransfer(getName(), getDestination());
086            bridgeTransfer.setMaximumTime(mMaximumTime);
087            super.build(bridgeTransfer);
088            return bridgeTransfer;
089
090        }
091    }
092
093    @Override
094    public int hashCode() {
095        final int prime = 31;
096        int result = super.hashCode();
097        result = prime * result + ((mMaximumTime == null) ? 0 : mMaximumTime.hashCode());
098        return result;
099    }
100
101    @Override
102    public boolean equals(Object obj) {
103        if (this == obj) return true;
104        if (!super.equals(obj)) return false;
105        if (getClass() != obj.getClass()) return false;
106        BridgeTransfer other = (BridgeTransfer) obj;
107        if (mMaximumTime == null) {
108            if (other.mMaximumTime != null) return false;
109        } else if (!mMaximumTime.equals(other.mMaximumTime)) return false;
110        return true;
111    }
112
113}
