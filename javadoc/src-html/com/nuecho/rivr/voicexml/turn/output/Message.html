005package com.nuecho.rivr.voicexml.turn.output;
006
007import static com.nuecho.rivr.core.util.Assert.*;
008import static com.nuecho.rivr.voicexml.rendering.voicexml.VoiceXmlDomUtil.*;
009
010import java.util.*;
011
012import javax.json.*;
013
014import org.w3c.dom.*;
015
016import com.nuecho.rivr.core.util.*;
017import com.nuecho.rivr.voicexml.dialogue.*;
018import com.nuecho.rivr.voicexml.rendering.voicexml.*;
019import com.nuecho.rivr.voicexml.turn.output.audio.*;
020import com.nuecho.rivr.voicexml.util.json.*;
021
022/**
023 * A {@link Message} is a {@link VoiceXmlOutputTurn} that plays a sequence of
024 * {@link AudioItem}.
025 * 
026 * @author Nu Echo Inc.
027 * @see AudioItem
028 * @see <a
029 *      href="http://www.w3.org/TR/voicexml20/#dml4.1.8">http://www.w3.org/TR/voicexml20/#dml4.1.8</a>
030 */
031public class Message extends VoiceXmlOutputTurn {
032    private static final String MESSAGE_TURN_TYPE = "message";
033
034    private static final String BARGE_IN_PROPERTY = "bargeIn";
035    private static final String LANGUAGE_PROPERTY = "language";
036    private static final String AUDIO_ITEMS_PROPERTY = "audioItems";
037
038    private final List<AudioItem> mAudioItems;
039    private String mLanguage;
040    private Boolean mBargeIn;
041
042    /**
043     * @param name The name of this turn. Not empty.
044     * @param audioItems The sequence of {@link AudioItem} to play. Not empty.
045     */
046    public Message(String name, List<AudioItem> audioItems) {
047        super(name);
048        Assert.notEmpty(audioItems, "audioItems");
049        mAudioItems = new ArrayList<AudioItem>(audioItems);
050    }
051
052    /**
053     * @param name The name of this turn. Not empty.
054     * @param audioItems The sequence of {@link AudioItem} to play. Not empty.
055     */
056    public Message(String name, AudioItem... audioItems) {
057        this(name, asListChecked(audioItems));
058    }
059
060    /**
061     * @param language The language identifier (e.g. "en-US") for the message.
062     *            <code>null</code> to use the VoiceXML platform default
063     */
064    public final void setLanguage(String language) {
065        mLanguage = language;
066    }
067
068    /**
069     * @param bargeIn <ul>
070     *            <li>{@link Boolean#TRUE} to enable barge-in</li>
071     *            <li>
072     *            {@link Boolean#FALSE} to disable barge-in</li>
073     *            <li><code>null</code> to use the VoiceXML platform default</li>
074     *            </ul>
075     */
076    public final void setBargeIn(Boolean bargeIn) {
077        mBargeIn = bargeIn;
078    }
079
080    public final List<AudioItem> getAudioItems() {
081        return Collections.unmodifiableList(mAudioItems);
082    }
083
084    public final String getLanguage() {
085        return mLanguage;
086    }
087
088    public final Boolean getBargeIn() {
089        return mBargeIn;
090    }
091
092    @Override
093    protected final String getOuputTurnType() {
094        return MESSAGE_TURN_TYPE;
095    }
096
097    @Override
098    protected void addTurnProperties(JsonObjectBuilder builder) {
099        JsonUtils.add(builder, AUDIO_ITEMS_PROPERTY, JsonUtils.toJson(mAudioItems));
100        JsonUtils.add(builder, LANGUAGE_PROPERTY, mLanguage);
101        if (mBargeIn == null) {
102            builder.addNull(BARGE_IN_PROPERTY);
103        } else {
104            builder.add(BARGE_IN_PROPERTY, mBargeIn.booleanValue());
105        }
106    }
107
108    @Override
109    protected void fillVoiceXmlDocument(Document document, Element formElement, VoiceXmlDialogueContext dialogueContext)
110            throws VoiceXmlDocumentRenderingException {
111        Element blockElement = addBlockElement(formElement);
112        createPrompt(mLanguage, blockElement, dialogueContext, mBargeIn, mAudioItems);
113        createGotoSubmit(blockElement);
114    }
115
116    /**
117     * Builder used to ease the creation of instances of {@link Message}.
118     */
119    public static class Builder {
120
121        private final String mName;
122        private final List<AudioItem> mAudioItems = new ArrayList<AudioItem>();
123        private String mLanguage;
124        private Boolean mBargeIn;
125
126        public Builder(String name) {
127            mName = name;
128        }
129
130        /**
131         * @deprecated Use {@link #addAudioItem(AudioItem)} instead
132         */
133        @Deprecated
134        public Builder addAudio(AudioItem audioItem) {
135            return addAudioItem(audioItem);
136        }
137
138        /**
139         * @since 1.0.1
140         */
141        public Builder addAudioItem(AudioItem audioItem) {
142            Assert.notNull(audioItem, "audioItem");
143            mAudioItems.add(audioItem);
144            return this;
145        }
146
147        /**
148         * @since 1.0.1
149         */
150        public Builder addAudioItems(AudioItem... audioItems) {
151            return addAudioItems(asListChecked(audioItems));
152        }
153
154        public Builder addAudioItems(List<AudioItem> audioItems) {
155            Assert.noNullValues(audioItems, mLanguage);
156            mAudioItems.addAll(audioItems);
157            return this;
158        }
159
160        public Builder setLanguage(String language) {
161            mLanguage = language;
162            return this;
163        }
164
165        public Builder setBargein(Boolean bargeIn) {
166            mBargeIn = bargeIn;
167            return this;
168        }
169
170        public Message build() {
171            Message message = new Message(mName, mAudioItems);
172            message.setBargeIn(mBargeIn);
173            message.setLanguage(mLanguage);
174            return message;
175        }
176    }
177}
