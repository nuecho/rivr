004package com.nuecho.rivr.voicexml.turn.output.audio;
005
006import javax.json.*;
007
008import com.nuecho.rivr.core.util.*;
009import com.nuecho.rivr.voicexml.turn.output.fetch.*;
010import com.nuecho.rivr.voicexml.util.json.*;
011
012/**
013 * A {@link AudioFile} represents an audio file with an optional synthesis text
014 * backup.
015 * 
016 * @author Nu Echo Inc.
017 * @see SpeechSynthesis
018 * @see FetchConfiguration
019 */
020public final class AudioFile extends AudioItem {
021    public static final String AUDIO_FILE_ELEMENT_TYPE = "audioFile";
022    private static final String FETCH_CONFIGURATION_PROPERTY = "fetchConfiguration";
023    private static final String LOCATION_PROPERTY = "location";
024    private static final String EXPRESSION_PROPERTY = "expression";
025    private static final String ALTERNATE_PROPERTY = "alternate";
026
027    private String mLocation;
028    private String mExpression;
029    private SpeechSynthesis mAlternate;
030    private FetchConfiguration mFetchConfiguration;
031
032    public static AudioFile fromLocation(String location) {
033        Assert.notEmpty(location, "location");
034        AudioFile audioFile = new AudioFile();
035        audioFile.mLocation = location;
036        return audioFile;
037    }
038
039    public static AudioFile fromExpression(String expression) {
040        Assert.notEmpty(expression, "expression");
041        AudioFile audioFile = new AudioFile();
042        audioFile.mExpression = expression;
043        return audioFile;
044    }
045
046    public static AudioFile fromLocation(String location, SpeechSynthesis alternate) {
047        AudioFile audioFile = fromLocation(location);
048        audioFile.setAlternate(alternate);
049        return audioFile;
050    }
051
052    public static AudioFile fromExpression(String expression, SpeechSynthesis alternate) {
053        AudioFile audioFile = fromExpression(expression);
054        audioFile.setAlternate(alternate);
055        return audioFile;
056    }
057
058    public static AudioFile fromLocation(String location, String alternate) {
059        AudioFile audioFile = fromLocation(location);
060        audioFile.setAlternate(alternate);
061        return audioFile;
062    }
063
064    public static AudioFile fromExpression(String expression, String alternate) {
065        AudioFile audioFile = fromExpression(expression);
066        audioFile.setAlternate(alternate);
067        return audioFile;
068    }
069
070    /**
071     * @param fetchConfiguration The resource fetch configuration.
072     *            <code>null</code> to use the VoiceXML platform default.
073     */
074    public void setFetchConfiguration(FetchConfiguration fetchConfiguration) {
075        mFetchConfiguration = fetchConfiguration;
076    }
077
078    public void setAlternate(SpeechSynthesis alternate) {
079        mAlternate = alternate;
080    }
081
082    public void setAlternate(String alternate) {
083        mAlternate = new SpeechSynthesis(alternate);
084    }
085
086    @Override
087    public String getElementType() {
088        return AUDIO_FILE_ELEMENT_TYPE;
089    }
090
091    public String getLocation() {
092        return mLocation;
093    }
094
095    public String getExpression() {
096        return mExpression;
097    }
098
099    public SpeechSynthesis getAlternate() {
100        return mAlternate;
101    }
102
103    public FetchConfiguration getFetchConfiguration() {
104        return mFetchConfiguration;
105    }
106
107    @Override
108    protected void addJsonProperties(JsonObjectBuilder builder) {
109        JsonUtils.add(builder, LOCATION_PROPERTY, mLocation);
110        JsonUtils.add(builder, EXPRESSION_PROPERTY, mExpression);
111        JsonUtils.add(builder, FETCH_CONFIGURATION_PROPERTY, mFetchConfiguration);
112        JsonUtils.add(builder, ALTERNATE_PROPERTY, mAlternate);
113    }
114
115    @Override
116    public int hashCode() {
117        final int prime = 31;
118        int result = 1;
119        result = prime * result + (mExpression == null ? 0 : mExpression.hashCode());
120        result = prime * result + (mLocation == null ? 0 : mLocation.hashCode());
121        result = prime * result + (mFetchConfiguration == null ? 0 : mFetchConfiguration.hashCode());
122        result = prime * result + (mAlternate == null ? 0 : mAlternate.hashCode());
123        return result;
124    }
125
126    @Override
127    public boolean equals(Object obj) {
128        if (this == obj) return true;
129        if (obj == null) return false;
130        if (getClass() != obj.getClass()) return false;
131        AudioFile other = (AudioFile) obj;
132        if (mExpression == null) {
133            if (other.mExpression != null) return false;
134        } else if (!mExpression.equals(other.mExpression)) return false;
135        if (mLocation == null) {
136            if (other.mLocation != null) return false;
137        } else if (!mLocation.equals(other.mLocation)) return false;
138        if (mFetchConfiguration == null) {
139            if (other.mFetchConfiguration != null) return false;
140        } else if (!mFetchConfiguration.equals(other.mFetchConfiguration)) return false;
141        if (mAlternate == null) {
142            if (other.mAlternate != null) return false;
143        } else if (!mAlternate.equals(other.mAlternate)) return false;
144        return true;
145    }
146
147}
