005package com.nuecho.rivr.voicexml.turn.output;
006
007import static com.nuecho.rivr.core.util.Assert.*;
008
009import java.util.*;
010import java.util.Map.Entry;
011
012import javax.json.*;
013
014import com.nuecho.rivr.core.util.*;
015import com.nuecho.rivr.voicexml.turn.output.grammar.*;
016import com.nuecho.rivr.voicexml.util.json.*;
017
018/**
019 * Base class of an interaction phase recognition configuration parts.
020 * 
021 * @author Nu Echo Inc.
022 * @see DtmfRecognition
023 * @see SpeechRecognition
024 */
025public abstract class Recognition implements JsonSerializable {
026    private static final String GRAMMARS_PROPERTY = "grammars";
027    private static final String PROPERTIES_PROPERTY = "properties";
028
029    private List<GrammarItem> mGrammarItems;
030    private final Map<String, String> mProperties = new HashMap<String, String>();
031
032    /**
033     * @param grammarItems The list of {@link GrammarItem}. Not null.
034     */
035    Recognition(List<? extends GrammarItem> grammarItems) {
036        setGrammarItems(grammarItems);
037    }
038
039    /**
040     * @param grammarItems The list of {@link GrammarItem}. Not null.
041     */
042    Recognition(GrammarItem... grammarItems) {
043        setGrammarItems(grammarItems);
044    }
045
046    /**
047     * Adds a property to the enclosing form
048     * 
049     * @param propertyName The name of the property. Not empty.
050     * @param propertyValue The value of the property. Not null.
051     * @see <a
052     *      href="https://www.w3.org/TR/voicexml20/#dml6.3">https://www.w3.org/TR/voicexml20/#dml6.3</a>
053     */
054    public void addProperty(String propertyName, String propertyValue) {
055        Assert.notEmpty(propertyName, "propertyName");
056        Assert.notNull(propertyValue, "propertyValue");
057        mProperties.put(propertyName, propertyValue);
058    }
059
060    public void removeProperty(String propertyName) {
061        mProperties.remove(propertyName);
062    }
063
064    public List<GrammarItem> getGrammarItems() {
065        return Collections.unmodifiableList(mGrammarItems);
066    }
067
068    public boolean hasProperty(String propertyName) {
069        return mProperties.containsKey(propertyName);
070    }
071
072    public Set<String> getPropertyNames() {
073        return Collections.unmodifiableSet(mProperties.keySet());
074    }
075
076    public String getProperty(String propertyName) {
077        return mProperties.get(propertyName);
078    }
079
080    protected final void copyPropertiesTo(Recognition copy) {
081        for (String propertyName : getPropertyNames()) {
082            copy.addProperty(propertyName, getProperty(propertyName));
083        }
084    }
085
086    public void setGrammarItems(GrammarItem... grammarItems) {
087        setGrammarItems(asListChecked(grammarItems));
088    }
089
090    public void setGrammarItems(List<? extends GrammarItem> grammarItems) {
091        Assert.noNullValues(grammarItems, "grammarItems");
092        mGrammarItems = new ArrayList<GrammarItem>(grammarItems);
093    }
094
095    @Override
096    public int hashCode() {
097        final int prime = 31;
098        int result = 1;
099        result = prime * result + ((mGrammarItems == null) ? 0 : mGrammarItems.hashCode());
100        result = prime * result + ((mProperties == null) ? 0 : mProperties.hashCode());
101        return result;
102    }
103
104    @Override
105    public boolean equals(Object obj) {
106        if (this == obj) return true;
107        if (obj == null) return false;
108        if (getClass() != obj.getClass()) return false;
109        Recognition other = (Recognition) obj;
110        if (mGrammarItems == null) {
111            if (other.mGrammarItems != null) return false;
112        } else if (!mGrammarItems.equals(other.mGrammarItems)) return false;
113        if (mProperties == null) {
114            if (other.mProperties != null) return false;
115        } else if (!mProperties.equals(other.mProperties)) return false;
116        return true;
117    }
118
119    @Override
120    public final String toString() {
121        return asJson().toString();
122    }
123
124    @Override
125    public JsonValue asJson() {
126        JsonObjectBuilder builder = JsonUtils.createObjectBuilder();
127
128        JsonObjectBuilder propertiesBuilder = JsonUtils.createObjectBuilder();
129        for (Entry<String, String> entry : mProperties.entrySet()) {
130            JsonUtils.add(propertiesBuilder, entry.getKey(), entry.getValue());
131        }
132        JsonUtils.add(builder, PROPERTIES_PROPERTY, propertiesBuilder);
133        JsonUtils.add(builder, GRAMMARS_PROPERTY, JsonUtils.toJson(mGrammarItems));
134        addJsonProperties(builder);
135        return builder.build();
136    }
137
138    protected abstract void addJsonProperties(JsonObjectBuilder builder);
139}
