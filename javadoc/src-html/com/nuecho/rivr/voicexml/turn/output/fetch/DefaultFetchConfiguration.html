005package com.nuecho.rivr.voicexml.turn.output.fetch;
006
007import static com.nuecho.rivr.voicexml.turn.output.fetch.ResourceDefaultFetchConfiguration.ResourceType.*;
008
009import com.nuecho.rivr.core.util.*;
010import com.nuecho.rivr.voicexml.dialogue.*;
011import com.nuecho.rivr.voicexml.turn.output.audio.*;
012
013/**
014 * Specifies the default properties for resource and document fetching. This
015 * class is used in the {@link VoiceXmlDialogueContext} and is used to generate
016 * a series of properties in each generated VoiceXML document.
017 * <p>
018 * To specify the fetch properties on a per-resource basis, specify a
019 * {@link FetchConfiguration} on the required resource, for example
020 * {@link AudioFile#setFetchConfiguration(FetchConfiguration)}.
021 * 
022 * @see VoiceXmlDialogueContext
023 * @author Nu Echo Inc.
024 */
025public final class DefaultFetchConfiguration {
026
027    private final ResourceDefaultFetchConfiguration mAudioFetchConfiguration = new ResourceDefaultFetchConfiguration(audio);
028    private final ResourceDefaultFetchConfiguration mGrammarFetchConfiguration = new ResourceDefaultFetchConfiguration(grammar);
029    private final ResourceDefaultFetchConfiguration mScriptFetchConfiguration = new ResourceDefaultFetchConfiguration(script);
030    private final ResourceDefaultFetchConfiguration mObjectFetchConfiguration = new ResourceDefaultFetchConfiguration(object);
031
032    private DocumentFetchConfiguration mDocumentFetchConfiguration = new DocumentFetchConfiguration();
033
034    private Duration mDefaultFetchTimeout;
035    private String mDefaultFetchAudio;
036
037    public ResourceDefaultFetchConfiguration getDefaultAudioFetchConfiguration() {
038        return mAudioFetchConfiguration;
039    }
040
041    public ResourceDefaultFetchConfiguration getDefaultGrammarFetchConfiguration() {
042        return mGrammarFetchConfiguration;
043    }
044
045    public ResourceDefaultFetchConfiguration getDefaultScriptFetchConfiguration() {
046        return mScriptFetchConfiguration;
047    }
048
049    public ResourceDefaultFetchConfiguration getDefaultObjectFetchConfiguration() {
050        return mObjectFetchConfiguration;
051    }
052
053    public Duration getDefaultFetchTimeout() {
054        return mDefaultFetchTimeout;
055    }
056
057    public void setDefaultFetchTimeout(Duration defaultFetchTimeout) {
058        mDefaultFetchTimeout = defaultFetchTimeout;
059    }
060
061    public String getDefaultFetchAudio() {
062        return mDefaultFetchAudio;
063    }
064
065    public void setDefaultFetchAudio(String defaultFetchAudio) {
066        mDefaultFetchAudio = defaultFetchAudio;
067    }
068
069    public DocumentFetchConfiguration getDocumentFetchConfiguration() {
070        return mDocumentFetchConfiguration;
071    }
072
073    public void setDocumentFetchConfiguration(DocumentFetchConfiguration documentFetchConfiguration) {
074        mDocumentFetchConfiguration = documentFetchConfiguration;
075    }
076
077    @Override
078    public int hashCode() {
079        final int prime = 31;
080        int result = 1;
081        result = prime * result + ((mAudioFetchConfiguration == null) ? 0 : mAudioFetchConfiguration.hashCode());
082        result = prime * result + ((mDefaultFetchAudio == null) ? 0 : mDefaultFetchAudio.hashCode());
083        result = prime * result + ((mDefaultFetchTimeout == null) ? 0 : mDefaultFetchTimeout.hashCode());
084        result = prime * result + ((mGrammarFetchConfiguration == null) ? 0 : mGrammarFetchConfiguration.hashCode());
085        result = prime * result + ((mObjectFetchConfiguration == null) ? 0 : mObjectFetchConfiguration.hashCode());
086        result = prime * result + ((mScriptFetchConfiguration == null) ? 0 : mScriptFetchConfiguration.hashCode());
087        result = prime * result + ((mDocumentFetchConfiguration == null) ? 0 : mDocumentFetchConfiguration.hashCode());
088        return result;
089    }
090
091    @Override
092    public boolean equals(Object obj) {
093        if (this == obj) return true;
094        if (obj == null) return false;
095        if (getClass() != obj.getClass()) return false;
096        DefaultFetchConfiguration other = (DefaultFetchConfiguration) obj;
097        if (mAudioFetchConfiguration == null) {
098            if (other.mAudioFetchConfiguration != null) return false;
099        } else if (!mAudioFetchConfiguration.equals(other.mAudioFetchConfiguration)) return false;
100        if (mDefaultFetchAudio == null) {
101            if (other.mDefaultFetchAudio != null) return false;
102        } else if (!mDefaultFetchAudio.equals(other.mDefaultFetchAudio)) return false;
103        if (mDefaultFetchTimeout == null) {
104            if (other.mDefaultFetchTimeout != null) return false;
105        } else if (!mDefaultFetchTimeout.equals(other.mDefaultFetchTimeout)) return false;
106        if (mGrammarFetchConfiguration == null) {
107            if (other.mGrammarFetchConfiguration != null) return false;
108        } else if (!mGrammarFetchConfiguration.equals(other.mGrammarFetchConfiguration)) return false;
109        if (mObjectFetchConfiguration == null) {
110            if (other.mObjectFetchConfiguration != null) return false;
111        } else if (!mObjectFetchConfiguration.equals(other.mObjectFetchConfiguration)) return false;
112        if (mScriptFetchConfiguration == null) {
113            if (other.mScriptFetchConfiguration != null) return false;
114        } else if (!mScriptFetchConfiguration.equals(other.mScriptFetchConfiguration)) return false;
115        if (mDocumentFetchConfiguration == null) {
116            if (other.mDocumentFetchConfiguration != null) return false;
117        } else if (!mDocumentFetchConfiguration.equals(other.mDocumentFetchConfiguration)) return false;
118        return true;
119    }
120
121    @Override
122    public String toString() {
123        ToStringBuilder builder = new ToStringBuilder(this);
124        builder.appendItem("mDocumentFetchConfiguration", mDocumentFetchConfiguration);
125        builder.appendItem("mDefaultFetchAudio", mDefaultFetchAudio);
126        builder.appendItem("mDefaultFetchTimeout", mDefaultFetchTimeout);
127        builder.appendItem("mAudioFetchConfiguration", mAudioFetchConfiguration);
128        builder.appendItem("mGrammarFetchConfiguration", mGrammarFetchConfiguration);
129        builder.appendItem("mObjectFetchConfiguration", mObjectFetchConfiguration);
130        builder.appendItem("mScriptFetchConfiguration", mScriptFetchConfiguration);
131        return builder.getString();
132    }
133}
