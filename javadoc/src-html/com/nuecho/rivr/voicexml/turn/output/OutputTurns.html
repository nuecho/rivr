005package com.nuecho.rivr.voicexml.turn.output;
006
007/**
008 * Factory methods for various <i>fluent</i> builders for
009 * {@link VoiceXmlOutputTurn VoiceXmlOutputTurns}.
010 * 
011 * @author Nu Echo Inc.
012 */
013public final class OutputTurns {
014
015    private OutputTurns() {
016        //utility class: instantiation forbidden
017    }
018
019    public static Interaction.Builder interaction(String name) {
020        return new Interaction.Builder(name);
021    }
022
023    public static Message.Builder message(String name) {
024        return new Message.Builder(name);
025    }
026
027    public static Script.Builder script(String name) {
028        return new Script.Builder(name);
029    }
030
031    public static ObjectCall.Builder objectCall(String name) {
032        return new ObjectCall.Builder(name);
033    }
034
035    public static SubdialogueCall.Builder subdialogueCall(String name) {
036        return new SubdialogueCall.Builder(name);
037    }
038
039    public static BlindTransfer.Builder blindTransfer(String name) {
040        return new BlindTransfer.Builder(name);
041    }
042
043    public static BridgeTransfer.Builder bridgeTransfer(String name) {
044        return new BridgeTransfer.Builder(name);
045    }
046
047    public static ConsultationTransfer.Builder consultationTransfer(String name) {
048        return new ConsultationTransfer.Builder(name);
049    }
050
051}
