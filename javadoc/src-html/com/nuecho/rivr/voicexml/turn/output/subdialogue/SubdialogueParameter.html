005    package com.nuecho.rivr.voicexml.turn.output.subdialogue;
006    
007    import javax.json.*;
008    
009    import com.nuecho.rivr.core.util.*;
010    import com.nuecho.rivr.voicexml.util.json.*;
011    
012    /**
013     * @author Nu Echo Inc.
014     */
015    public final class SubdialogueParameter implements JsonSerializable {
016        private static final String NAME_PROPERTY = "name";
017        private static final String VALUE_PROPERTY = "value";
018        private static final String EXPRESSION_PROPERTY = "expression";
019    
020        private final String mName;
021        private String mExpression;
022        private String mValue;
023    
024        /**
025         * @param name The name of the parameter. Not empty.
026         * @param value The string value of the parameter. Not null.
027         * @return The newly created subdialogue parameter
028         */
029        public static SubdialogueParameter createWithValue(String name, String value) {
030            Assert.notNull(value, "value");
031    
032            SubdialogueParameter parameter = new SubdialogueParameter(name);
033            parameter.mValue = value;
034            return parameter;
035        }
036    
037        /**
038         * @param name The name of the parameter. Not empty.
039         * @param json The JSON value of the parameter. Not null.
040         * @return The newly created subdialogue parameter
041         */
042        public static SubdialogueParameter createWithJson(String name, JsonValue json) {
043            Assert.notNull(json, "json");
044            return createWithExpression(name, json.toString());
045        }
046    
047        /**
048         * @param name The name of the parameter. Not empty.
049         * @param expression The ECMAScript expression of the parameter. Not null.
050         * @return The newly created subdialogue parameter
051         */
052        public static SubdialogueParameter createWithExpression(String name, String expression) {
053            Assert.notNull(expression, "expression");
054    
055            SubdialogueParameter parameter = new SubdialogueParameter(name);
056            parameter.mExpression = expression;
057            return parameter;
058        }
059    
060        private SubdialogueParameter(String name) {
061            Assert.notEmpty(name, "name");
062            mName = name;
063        }
064    
065        public String getName() {
066            return mName;
067        }
068    
069        public String getExpression() {
070            return mExpression;
071        }
072    
073        public String getValue() {
074            return mValue;
075        }
076    
077        @Override
078        public JsonValue asJson() {
079            JsonObjectBuilder builder = JsonUtils.createObjectBuilder();
080            JsonUtils.add(builder, NAME_PROPERTY, mName);
081            JsonUtils.add(builder, EXPRESSION_PROPERTY, mExpression);
082            JsonUtils.add(builder, VALUE_PROPERTY, mValue);
083            return builder.build();
084        }
085    }
