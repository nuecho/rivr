005    package com.nuecho.rivr.voicexml.turn.output.fetch;
006    
007    import javax.json.*;
008    
009    import com.nuecho.rivr.voicexml.turn.output.audio.*;
010    import com.nuecho.rivr.voicexml.util.json.*;
011    
012    /**
013     * @author Nu Echo Inc.
014     */
015    public final class SubdialogueFetchConfiguration implements JsonSerializable {
016    
017        private static final String RESOURCE_FETCH_CONFIGURATION = "resourceFetchConfiguration";
018        private static final String FETCH_AUDIO = "fetchAudio";
019    
020        private ResourceFetchConfiguration mResourceFetchConfiguration;
021        private Recording mFetchAudio;
022    
023        public ResourceFetchConfiguration getResourceFetchConfiguration() {
024            return mResourceFetchConfiguration;
025        }
026    
027        public Recording getFetchAudio() {
028            return mFetchAudio;
029        }
030    
031        public void setResourceFetchConfiguration(ResourceFetchConfiguration resourceFetchConfiguration) {
032            mResourceFetchConfiguration = resourceFetchConfiguration;
033        }
034    
035        public void setFetchAudio(Recording fetchAudio) {
036            mFetchAudio = fetchAudio;
037        }
038    
039        @Override
040        public JsonValue asJson() {
041            JsonObjectBuilder builder = JsonUtils.createObjectBuilder();
042            JsonUtils.add(builder, FETCH_AUDIO, mFetchAudio);
043            JsonUtils.add(builder, RESOURCE_FETCH_CONFIGURATION, mResourceFetchConfiguration);
044            return builder.build();
045        }
046    }
