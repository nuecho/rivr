004package com.nuecho.rivr.voicexml.turn.output.grammar;
005
006import javax.json.*;
007
008import com.nuecho.rivr.core.util.*;
009import com.nuecho.rivr.voicexml.turn.output.fetch.*;
010import com.nuecho.rivr.voicexml.util.json.*;
011
012/**
013 * A {@link GrammarReference} represents an external grammar reachable with an
014 * URI.
015 * 
016 * @author Nu Echo Inc.
017 * @see <a
018 *      href="https://www.w3.org/TR/voicexml20/#dml3.1.1.2">https://www.w3.org/TR/voicexml20/#dml3.1.1.2</a>
019 */
020public final class GrammarReference extends GrammarItem {
021    private static final String REFERENCE_ELEMENT_TYPE = "reference";
022    private static final String URI_PROPERTY = "uri";
023    private static final String FETCH_CONFIGURATION_PROPERTY = "fetchConfiguration";
024
025    private final String mUri;
026    private FetchConfiguration mFetchConfiguration;
027
028    /**
029     * @param uri The URI of the external grammar. Not empty.
030     */
031    public GrammarReference(String uri) {
032        Assert.notEmpty(uri, "uri");
033        mUri = uri;
034    }
035
036    /**
037     * @param fetchConfiguration The resource fetch configuration.
038     *            <code>null</code> to use the VoiceXML platform default.
039     */
040    public void setFetchConfiguration(FetchConfiguration fetchConfiguration) {
041        mFetchConfiguration = fetchConfiguration;
042    }
043
044    @Override
045    public String getElementType() {
046        return REFERENCE_ELEMENT_TYPE;
047    }
048
049    public String getUri() {
050        return mUri;
051    }
052
053    public FetchConfiguration getFetchConfiguration() {
054        return mFetchConfiguration;
055    }
056
057    @Override
058    protected void addJsonProperties(JsonObjectBuilder builder) {
059        JsonUtils.add(builder, URI_PROPERTY, mUri);
060        JsonUtils.add(builder, FETCH_CONFIGURATION_PROPERTY, mFetchConfiguration);
061    }
062
063    @Override
064    public int hashCode() {
065        final int prime = 31;
066        int result = 1;
067        result = prime * result + (mFetchConfiguration == null ? 0 : mFetchConfiguration.hashCode());
068        result = prime * result + (mUri == null ? 0 : mUri.hashCode());
069        return result;
070    }
071
072    @Override
073    public boolean equals(Object obj) {
074        if (this == obj) return true;
075        if (obj == null) return false;
076        if (getClass() != obj.getClass()) return false;
077        GrammarReference other = (GrammarReference) obj;
078        if (mFetchConfiguration == null) {
079            if (other.mFetchConfiguration != null) return false;
080        } else if (!mFetchConfiguration.equals(other.mFetchConfiguration)) return false;
081        if (mUri == null) {
082            if (other.mUri != null) return false;
083        } else if (!mUri.equals(other.mUri)) return false;
084        return true;
085    }
086}
