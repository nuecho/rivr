005package com.nuecho.rivr.voicexml.turn.output;
006
007import java.util.*;
008
009import javax.json.*;
010
011import com.nuecho.rivr.core.util.*;
012import com.nuecho.rivr.voicexml.turn.output.grammar.*;
013import com.nuecho.rivr.voicexml.util.json.*;
014
015/**
016 * A {@link DtmfRecognition} represents the DTMF part of an interaction phase
017 * recognition configuration. It is composed of a list of DTMF
018 * {@link GrammarItem} and optional properties (inter-digit timeout, term
019 * timeout, term char).
020 * 
021 * @author Nu Echo Inc.
022 */
023public final class DtmfRecognition extends Recognition {
024    private static final String INTERDIGIT_TIMEOUT_PROPERTY = "interdigitTimeout";
025    private static final String TERM_TIMEOUT_PROPERTY = "termTimeout";
026    private static final String TERM_CHAR_PROPERTY = "termChar";
027
028    private Duration mInterDigitTimeout;
029    private Duration mTermTimeout;
030    private String mTermChar;
031
032    /**
033     * @param grammarItems The list of DTMF {@link GrammarItem}. Not null.
034     */
035    public DtmfRecognition(GrammarItem... grammarItems) {
036        super(grammarItems);
037    }
038
039    /**
040     * @param grammarItems The list of DTMF {@link GrammarItem}. Not null.
041     */
042    public DtmfRecognition(List<GrammarItem> grammarItems) {
043        super(grammarItems);
044    }
045
046    /**
047     * @param interDigitTimeout The inter-digit timeout value to use when
048     *            recognizing DTMF input. <code>null</code> to use the VoiceXML
049     *            platform default.
050     * @see <a
051     *      href="http://www.w3.org/TR/voicexml20/#dml6.3.3">http://www.w3.org/TR/voicexml20/#dml6.3.3</a>
052     */
053    public void setInterDigitTimeout(Duration interDigitTimeout) {
054        mInterDigitTimeout = interDigitTimeout;
055    }
056
057    /**
058     * @param termTimeout The terminating timeout to use when recognizing DTMF
059     *            input. <code>null</code> to use the VoiceXML platform default
060     * @see <a
061     *      href="http://www.w3.org/TR/voicexml20/#dml6.3.3">http://www.w3.org/TR/voicexml20/#dml6.3.3</a>
062     */
063    public void setTermTimeout(Duration termTimeout) {
064        mTermTimeout = termTimeout;
065    }
066
067    /**
068     * @param termChar The terminating DTMF character for DTMF input
069     *            recognition. Must be empty or a single character.
070     *            <code>null</code> to use the VoiceXML platform default
071     * @see <a
072     *      href="http://www.w3.org/TR/voicexml20/#dml6.3.3">http://www.w3.org/TR/voicexml20/#dml6.3.3</a>
073     */
074    public void setTermChar(String termChar) {
075        if (termChar != null) {
076            Assert.ensure(termChar.equals("") || termChar.length() == 1, "Termchar must be empty or single character");
077        }
078
079        mTermChar = termChar;
080    }
081
082    public Duration getInterDigitTimeout() {
083        return mInterDigitTimeout;
084    }
085
086    public Duration getTermTimeout() {
087        return mTermTimeout;
088    }
089
090    public String getTermChar() {
091        return mTermChar;
092    }
093
094    @Override
095    protected void addJsonProperties(JsonObjectBuilder builder) {
096        JsonUtils.addDurationProperty(builder, INTERDIGIT_TIMEOUT_PROPERTY, mInterDigitTimeout);
097        JsonUtils.addDurationProperty(builder, TERM_TIMEOUT_PROPERTY, mTermTimeout);
098        JsonUtils.add(builder, TERM_CHAR_PROPERTY, mTermChar);
099    }
100
101    @Override
102    public int hashCode() {
103        final int prime = 31;
104        int result = super.hashCode();
105        result = prime * result + (mInterDigitTimeout == null ? 0 : mInterDigitTimeout.hashCode());
106        result = prime * result + (mTermChar == null ? 0 : mTermChar.hashCode());
107        result = prime * result + (mTermTimeout == null ? 0 : mTermTimeout.hashCode());
108        return result;
109    }
110
111    @Override
112    public boolean equals(Object obj) {
113        if (this == obj) return true;
114        if (!super.equals(obj)) return false;
115        if (getClass() != obj.getClass()) return false;
116        DtmfRecognition other = (DtmfRecognition) obj;
117        if (mInterDigitTimeout == null) {
118            if (other.mInterDigitTimeout != null) return false;
119        } else if (!mInterDigitTimeout.equals(other.mInterDigitTimeout)) return false;
120        if (mTermChar == null) {
121            if (other.mTermChar != null) return false;
122        } else if (!mTermChar.equals(other.mTermChar)) return false;
123        if (mTermTimeout == null) {
124            if (other.mTermTimeout != null) return false;
125        } else if (!mTermTimeout.equals(other.mTermTimeout)) return false;
126        return true;
127    }
128
129    public DtmfRecognition copy() {
130        DtmfRecognition copy = new DtmfRecognition(getGrammarItems());
131        copy.mInterDigitTimeout = mInterDigitTimeout;
132        copy.mTermChar = mTermChar;
133        copy.mTermTimeout = mTermTimeout;
134        copyPropertiesTo(copy);
135        return copy;
136    }
137}
