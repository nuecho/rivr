004package com.nuecho.rivr.voicexml.turn.input;
005
006import javax.json.*;
007
008import com.nuecho.rivr.voicexml.turn.output.*;
009import com.nuecho.rivr.voicexml.util.json.*;
010
011/**
012 * The status of a {@link Transfer}.
013 * 
014 * @author NuEcho Inc.
015 * @see <a
016 *      href="http://www.w3.org/TR/voicexml20/#dml2.3.7.2.2">http://www.w3.org/TR/voicexml20/#dml2.3.7.2.2</a>
017 */
018public class TransferStatus implements JsonSerializable {
019
020    public static final String NO_ANSWER = "noanswer";
021    public static final String NEAR_END_DISCONNECT = "near_end_disconnect";
022    public static final String FAR_END_DISCONNECT = "far_end_disconnect";
023    public static final String NETWORK_DISONNECT = "network_disconnect";
024    public static final String MAXTIME_DISCONNECT = "maxtime_disconnect";
025    public static final String BUSY = "busy";
026    public static final String NETWORK_BUSY = "network_busy";
027    public static final String UNKNOWN = "unknown";
028
029    private final String mStatusCode;
030
031    public TransferStatus(String statusCode) {
032        mStatusCode = statusCode;
033    }
034
035    public String getStatusCode() {
036        return mStatusCode;
037    }
038
039    @Override
040    public JsonValue asJson() {
041        return JsonUtils.wrap(mStatusCode);
042    }
043
044    @Override
045    public int hashCode() {
046        final int prime = 31;
047        int result = 1;
048        result = prime * result + ((mStatusCode == null) ? 0 : mStatusCode.hashCode());
049        return result;
050    }
051
052    @Override
053    public boolean equals(Object obj) {
054        if (this == obj) return true;
055        if (obj == null) return false;
056        if (getClass() != obj.getClass()) return false;
057        TransferStatus other = (TransferStatus) obj;
058        if (mStatusCode == null) {
059            if (other.mStatusCode != null) return false;
060        } else if (!mStatusCode.equals(other.mStatusCode)) return false;
061        return true;
062    }
063
064}
