005    package com.nuecho.rivr.voicexml.turn.input;
006    
007    import java.util.*;
008    
009    import javax.json.*;
010    
011    import com.nuecho.rivr.core.util.*;
012    import com.nuecho.rivr.voicexml.util.json.*;
013    
014    /**
015     * @author Nu Echo Inc.
016     */
017    public final class VoiceXmlEvent implements JsonSerializable {
018    
019        private static final String MESSAGE_PROPERT = "message";
020        private static final String NAME_PROPERTY = "name";
021    
022        private final String mName;
023        private final String mMessage;
024    
025        public static final String CANCEL = "cancel";
026        public static final String CONNECTION_DISCONNECT = "connection.disconnect";
027        public static final String CONNECTION_DISCONNECT_HANGUP = "connection.disconnect.hangup";
028        public static final String CONNECTION_DISCONNECT_TRANSFER = "connection.disconnect.transfer";
029        public static final String EXIT = "exit";
030        public static final String HELP = "help";
031        public static final String NO_INPUT = "noinput";
032        public static final String NO_MATCH = "nomatch";
033        public static final String MAX_SPEECH_TIMEOUT = "maxspeechtimeout";
034        public static final String ERROR = "error";
035        public static final String ERROR_BAD_FETCH = "error.badfetch";
036        public static final String ERROR_BAD_FETCH_HTTP = "error.badfetch.http";
037        public static final String ERROR_BAD_FETCH_PROTOCOL = "error.badfetch.protocol";
038        public static final String ERROR_SEMANTIC = "error.semantic";
039        public static final String ERROR_NO_AUTHORIZATION = "error.noauthorization";
040        public static final String ERROR_NO_RESOURCE = "error.noresource";
041        public static final String ERROR_UNSUPPORTED_BUILTIN = "error.unsupported.builtin";
042        public static final String ERROR_UNSUPPORTED_FORMAT = "error.unsupported.format";
043        public static final String ERROR_UNSUPPORTED_LANGUAGE = "error.unsupported.language";
044        public static final String ERROR_UNSUPPORTED_OBJECT_NAME = "error.unsupported.objectname";
045        public static final String ERROR_UNSUPPORTED = "error.unsupported";
046    
047        public static final String ERROR_CONNECTION = "error.connection";
048        public static final String ERROR_CONNECTION_NO_AUTHORIZATION = "error.connection.noauthorization";
049        public static final String ERROR_CONNECTION_BAD_DESTINATION = "error.connection.baddestination";
050        public static final String ERROR_CONNECTION_NO_ROUTE = "error.connection.noroute";
051        public static final String ERROR_CONNECTION_NO_RESOURCE = "error.connection.noresource";
052        public static final String ERROR_CONNECTION_PROTOCOL = "error.connection.protocol";
053        public static final String ERROR_UNSUPPORTED_TRANSFER_BLIND = "error.unsupported.transfer.blind";
054        public static final String ERROR_UNSUPPORTED_TRANSFER_BRIDGE = "error.unsupported.transfer.bridge";
055        public static final String ERROR_UNSUPPORTED_TRANSFER_CONSULTATION = "error.unsupported.transfer.consultation";
056        public static final String ERROR_UNSUPPORTED_URI = "error.unsupported.uri";
057    
058        public static boolean hasEvent(String prefix, Collection<? extends VoiceXmlEvent> events) {
059            for (VoiceXmlEvent voiceXmlEvent : events) {
060                if (voiceXmlEvent.isSubtypeOf(prefix)) return true;
061            }
062            return false;
063        }
064    
065        public VoiceXmlEvent(String name, String message) {
066            Assert.notEmpty(name, NAME_PROPERTY);
067            mName = name;
068            mMessage = message;
069        }
070    
071        public String getName() {
072            return mName;
073        }
074    
075        public String getMessage() {
076            return mMessage;
077        }
078    
079        public boolean isSubtypeOf(String prefix) {
080            Assert.notNull(prefix, "prefix");
081            return mName.equals(prefix) || mName.startsWith(prefix + ".");
082        }
083    
084        @Override
085        public String toString() {
086            return asJson().toString();
087        }
088    
089        @Override
090        public JsonValue asJson() {
091            JsonObjectBuilder builder = JsonUtils.createObjectBuilder();
092            JsonUtils.add(builder, NAME_PROPERTY, mName);
093            JsonUtils.add(builder, MESSAGE_PROPERT, mMessage);
094            return builder.build();
095        }
096    }
