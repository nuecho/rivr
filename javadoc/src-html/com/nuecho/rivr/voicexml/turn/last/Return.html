021/**
022 * Terminates a {@link VoiceXmlDialogue} with a return, transferring the control
023 * back to the calling dialogue. A list of variables or an event can be
024 * returned. Must be used it the {@link VoiceXmlDialogue} was called using a
025 * {@link SubdialogueCall}.
026 * 
027 * @author Nu Echo Inc.
028 * @see VoiceXmlDialogue
029 * @see LastTurn
030 * @see SubdialogueCall
031 * @see <a
032 *      href="http://www.w3.org/TR/voicexml20/#dml5.3.10">http://www.w3.org/TR/voicexml20/#dml5.3.10</a>
033 */
034public class Return extends VoiceXmlLastTurn {
035
036    private static final String VARIABLES_PROPERTY = "variables";
037    private static final String EVENT_MESSAGE_PROPERTY = "eventMessage";
038    private static final String EVENT_NAME_PROPERTY = "eventName";
039
040    private VariableList mVariables;
041    private String mEventName;
042    private String mEventMessage;
043
044    public Return(String name, VariableList variables) {
045        super(name);
046        Assert.notNull(variables, VARIABLES_PROPERTY);
047        mVariables = variables;
048    }
049
050    public Return(String name, String eventName, String eventMessage) {
051        super(name);
052        Assert.notNull(eventName, EVENT_NAME_PROPERTY);
053        mEventName = eventName;
054        mEventMessage = eventMessage;
055    }
056
057    public VariableList getVariables() {
058        return mVariables;
059    }
060
061    public String getEventName() {
062        return mEventName;
063    }
064
065    public String getEventMessage() {
066        return mEventMessage;
067    }
068
069    @Override
070    protected void addTurnProperties(JsonObjectBuilder builder) {
071        JsonUtils.add(builder, VARIABLES_PROPERTY, mVariables);
072        JsonUtils.add(builder, EVENT_NAME_PROPERTY, mEventName);
073        JsonUtils.add(builder, EVENT_MESSAGE_PROPERTY, mEventMessage);
074    }
075
076    @Override
077    protected void fillVoiceXmlDocument(Document document, Element formElement, VoiceXmlDialogueContext dialogueContext)
078            throws VoiceXmlDocumentRenderingException {
079        Element blockElement = DomUtils.appendNewElement(formElement, BLOCK_ELEMENT);
080        Element returnElement = document.createElement(RETURN_ELEMENT);
081
082        if (mVariables != null) {
083            addNamelist(blockElement, returnElement, mVariables);
084        } else if (mEventName != null) {
085            returnElement.setAttribute(EVENT_ATTRIBUTE, mEventName);
086
087            if (mEventMessage != null) {
088                returnElement.setAttribute(MESSAGE_ATTRIBUTE, mEventMessage);
089            }
090        }
091        blockElement.appendChild(returnElement);
092    }
093
094    @Override
095    public int hashCode() {
096        final int prime = 31;
097        int result = super.hashCode();
098        result = prime * result + ((mEventMessage == null) ? 0 : mEventMessage.hashCode());
099        result = prime * result + ((mEventName == null) ? 0 : mEventName.hashCode());
100        result = prime * result + ((mVariables == null) ? 0 : mVariables.hashCode());
101        return result;
102    }
103
104    @Override
105    public boolean equals(Object obj) {
106        if (this == obj) return true;
107        if (!super.equals(obj)) return false;
108        if (getClass() != obj.getClass()) return false;
109        Return other = (Return) obj;
110        if (mEventMessage == null) {
111            if (other.mEventMessage != null) return false;
112        } else if (!mEventMessage.equals(other.mEventMessage)) return false;
113        if (mEventName == null) {
114            if (other.mEventName != null) return false;
115        } else if (!mEventName.equals(other.mEventName)) return false;
116        if (mVariables == null) {
117            if (other.mVariables != null) return false;
118        } else if (!mVariables.equals(other.mVariables)) return false;
119        return true;
120    }
121
122}
