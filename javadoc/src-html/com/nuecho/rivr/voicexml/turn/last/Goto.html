020/**
021 * Terminates a {@link VoiceXmlDialogue} with a goto, transferring the control
022 * to another application specified with a URI.
023 * 
024 * @author Nu Echo Inc.
025 * @see VoiceXmlDialogue
026 * @see LastTurn
027 * @see <a
028 *      href="http://www.w3.org/TR/voicexml20/#dml5.3.7">http://www.w3.org/TR/voicexml20/#dml5.3.7</a>
029 */
030public class Goto extends VoiceXmlLastTurn {
031
032    private static final String FETCH_CONFIGURATION_PROPERTY = "fetchConfiguration";
033    private static final String URI_PROPERTY = "uri";
034
035    private final String mUri;
036    private DocumentFetchConfiguration mFetchConfiguration;
037
038    public Goto(String name, String uri) {
039        super(name);
040        Assert.notNull(uri, "uri");
041        mUri = uri;
042    }
043
044    public DocumentFetchConfiguration getFetchConfiguration() {
045        return mFetchConfiguration;
046    }
047
048    public void setFetchConfiguration(DocumentFetchConfiguration fetchConfiguration) {
049        mFetchConfiguration = fetchConfiguration;
050    }
051
052    public String getUri() {
053        return mUri;
054    }
055
056    @Override
057    protected void addTurnProperties(JsonObjectBuilder builder) {
058        JsonUtils.add(builder, URI_PROPERTY, mUri);
059        JsonUtils.add(builder, FETCH_CONFIGURATION_PROPERTY, mFetchConfiguration);
060    }
061
062    @Override
063    protected void fillVoiceXmlDocument(Document document, Element formElement, VoiceXmlDialogueContext dialogueContext)
064            throws VoiceXmlDocumentRenderingException {
065        Element blockElement = DomUtils.appendNewElement(formElement, BLOCK_ELEMENT);
066        Element submitElement = DomUtils.appendNewElement(blockElement, SUBMIT_ELEMENT);
067
068        VoiceXmlDomUtil.setAttribute(submitElement, VoiceXmlDomUtil.NEXT_ATTRIBUTE, mUri);
069
070        if (mFetchConfiguration != null) {
071            VoiceXmlDomUtil.applyDocumentFetchConfiguration(submitElement, dialogueContext);
072        }
073    }
074
075    @Override
076    public int hashCode() {
077        final int prime = 31;
078        int result = super.hashCode();
079        result = prime * result + ((mFetchConfiguration == null) ? 0 : mFetchConfiguration.hashCode());
080        result = prime * result + ((mUri == null) ? 0 : mUri.hashCode());
081        return result;
082    }
083
084    @Override
085    public boolean equals(Object obj) {
086        if (this == obj) return true;
087        if (!super.equals(obj)) return false;
088        if (getClass() != obj.getClass()) return false;
089        Goto other = (Goto) obj;
090        if (mFetchConfiguration == null) {
091            if (other.mFetchConfiguration != null) return false;
092        } else if (!mFetchConfiguration.equals(other.mFetchConfiguration)) return false;
093        if (mUri == null) {
094            if (other.mUri != null) return false;
095        } else if (!mUri.equals(other.mUri)) return false;
096        return true;
097    }
098
099}
